23class GymratLifter(Lifter): 

24 """ 

25 This is a base class for lifters that use Gymrat. 

26 For most architectures, all you need to do is subclass this, and set the property "instructions" 

27 to be a list of classes that define each instruction. 

28 By default, a lifter will decode instructions by attempting to instantiate every class until one works. 

29 This will use an IRSBCustomizer, which will, if it succeeds, add the appropriate VEX instructions to a pyvex IRSB. 

30 pyvex, when lifting a block of code for this architecture, will call the method "lift", which will produce the IRSB 

31 of the lifted code. 

32 """

48 def _decode_next_instruction(self, addr): 

49 # Try every instruction until one works 

50 for possible_instr in self.instrs: 

51 try: 

52 l.info("Trying " + possible_instr.name) 

53 return possible_instr(self.bitstrm, self.irsb.arch, addr) 

54 # a ParserError signals that this instruction did not match 

55 # we need to try other instructions, so we ignore this error 

56 except ParseError: 

57 pass #l.exception(repr(possible_instr)) 

58 # if we are out of input, ignore. 

59 # there may be other, shorter instructions that still match, 

60 # so we continue with the loop 

61 except bitstring.ReadError: 

62 pass 

63 

64 # If no instruction matches, log an error 

65 errorstr = 'Unknown instruction at bit position %d' % self.bitstrm.bitpos 

66 l.debug(errorstr) 

67 l.debug("Address: %#08x" % addr)

69 def decode(self): 

70 try: 

71 self.create_bitstrm() 

72 count = 0 

73 disas = [] 

74 addr = self.irsb.addr 

75 l.debug("Starting block at address: " + hex(addr)) 

76 bytepos = self.bitstrm.bytepos 

77 

78 

79 while (not is_empty(self.bitstrm)): 

80 instr = self._decode_next_instruction(addr) 

81 if not instr: break 

82 disas.append(instr) 

83 l.debug("Matched " + instr.name) 

84 addr += self.bitstrm.bytepos - bytepos 

85 bytepos = self.bitstrm.bytepos 

86 count += 1 

87 return disas 

88 except Exception as e: 

89 self.errors = str(e) 

90 l.exception("Error decoding block at offset {:#x} (address {:#x}):".format(bytepos, addr)) 

91 raise

94 self.thedata = self.data[:self.max_bytes] if isinstance(self.data, (bytes, bytearray, memoryview)) else self.data[:self.max_bytes].encode()

102 l.debug("Decoding complete.")

104 l.debug("Lifting instruction " + instr.name) 

105 instr(irsb_c, instructions[:i], instructions[i+1:]) 

106 if irsb_c.irsb.jumpkind != JumpKind.Invalid: 

107 break 

108 elif (i+1) == self.max_inst: # if we are on our last iteration
