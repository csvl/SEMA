18# AttributeValue - describes an attribute value in the DIE: 

19# 

20# name: 

21# The name (DW_AT_*) of this attribute 

22# 

23# form: 

24# The DW_FORM_* name of this attribute 

25# 

26# value: 

27# The value parsed from the section and translated accordingly to the form 

28# (e.g. for a DW_FORM_strp it's the actual string taken from the string table) 

29# 

30# raw_value: 

31# Raw value as parsed from the section - used for debugging and presentation 

32# (e.g. for a DW_FORM_strp it's the raw string offset into the table) 

33# 

34# offset: 

35# Offset of this attribute's value in the stream (absolute offset, relative 

36# the beginning of the whole stream) 

37# 

38# indirection_length: 

39# If the form of the attribute is DW_FORM_indirect, the form will contain 

40# the resolved form, and this will contain the length of the indirection chain. 

41# 0 means no indirection. 

42AttributeValue = namedtuple( 

43 'AttributeValue', 'name form value raw_value offset indirection_length')

46class DIE(object): 

47 """ A DWARF debugging information entry. On creation, parses itself from 

48 the stream. Each DIE is held by a CU. 

49 

50 Accessible attributes: 

51 

52 tag: 

53 The DIE tag 

54 

55 size: 

56 The size this DIE occupies in the section 

57 

58 offset: 

59 The offset of this DIE in the stream 

60 

61 attributes: 

62 An ordered dictionary mapping attribute names to values. It's 

63 ordered to preserve the order of attributes in the section 

64 

65 has_children: 

66 Specifies whether this DIE has children 

67 

68 abbrev_code: 

69 The abbreviation code pointing to an abbreviation entry (note 

70 that this is for informational purposes only - this object 

71 interacts with its abbreviation table transparently). 

72 

73 See also the public methods. 

74 """ 

75 def __init__(self, cu, stream, offset): 

76 """ cu: 

77 CompileUnit object this DIE belongs to. Used to obtain context 

78 information (structs, abbrev table, etc.) 

79 

80 stream, offset: 

81 The stream and offset into it where this DIE's data is located 

82 """ 

83 self.cu = cu 

84 self.dwarfinfo = self.cu.dwarfinfo # get DWARFInfo context 

85 self.stream = stream 

86 self.offset = offset 

87 

88 self.attributes = OrderedDict() 

89 self.tag = None 

90 self.has_children = None 

91 self.abbrev_code = None 

92 self.size = 0 

93 # Null DIE terminator. It can be used to obtain offset range occupied 

94 # by this DIE including its whole subtree. 

95 self._terminator = None 

96 self._parent = None 

97 

98 self._parse_DIE()

105 def get_DIE_from_attribute(self, name): 

106 """ Return the DIE referenced by the named attribute of this DIE. 

107 The attribute must be in the reference attribute class. 

108 

109 name: 

110 The name of the attribute in the reference class. 

111 """ 

112 attr = self.attributes[name] 

113 if attr.form in ('DW_FORM_ref1', 'DW_FORM_ref2', 'DW_FORM_ref4', 

114 'DW_FORM_ref8', 'DW_FORM_ref', 'DW_FORM_ref_udata'): 

115 refaddr = self.cu.cu_offset + attr.raw_value 

116 return self.cu.get_DIE_from_refaddr(refaddr) 

117 elif attr.form in ('DW_FORM_ref_addr'): 

118 return self.cu.dwarfinfo.get_DIE_from_refaddr(attr.raw_value) 

119 elif attr.form in ('DW_FORM_ref_sig8'): 

120 # Implement search type units for matching signature 

121 raise NotImplementedError('%s (type unit by signature)' % attr.form) 

122 elif attr.form in ('DW_FORM_ref_sup4', 'DW_FORM_ref_sup8', 'DW_FORM_GNU_ref_alt'): 

123 if self.dwarfinfo.supplementary_dwarfinfo: 

124 return self.dwarfinfo.supplementary_dwarfinfo.get_DIE_from_refaddr(attr.raw_value) 

125 # FIXME: how to distinguish supplementary files from dwo ? 

126 raise NotImplementedError('%s to dwo' % attr.form) 

127 else: 

128 raise DWARFError('%s is not a reference class form attribute' % attr)

130 def get_parent(self): 

131 """ Return the parent DIE of this DIE, or None if the DIE has no 

132 parent (i.e. is a top-level DIE). 

133 """ 

134 if self._parent is None: 

135 self._search_ancestor_offspring()

138 def get_full_path(self): 

139 """ Return the full path filename for the DIE. 

140 

141 The filename is the join of 'DW_AT_comp_dir' and 'DW_AT_name', 

142 either of which may be missing in practice. Note that its value is 

143 usually a string taken from the .debug_string section and the 

144 returned value will be a string. 

145 """ 

146 comp_dir_attr = self.attributes.get('DW_AT_comp_dir', None) 

147 comp_dir = bytes2str(comp_dir_attr.value) if comp_dir_attr else '' 

148 fname_attr = self.attributes.get('DW_AT_name', None) 

149 fname = bytes2str(fname_attr.value) if fname_attr else '' 

150 return os.path.join(comp_dir, fname)

157 def iter_siblings(self): 

158 """ Yield all siblings of this DIE 

159 """ 

160 parent = self.get_parent() 

161 if parent: 

162 for sibling in parent.iter_children(): 

163 if sibling is not self: 

164 yield sibling 

165 else: 

166 raise StopIteration()

177 def _search_ancestor_offspring(self): 

178 """ Search our ancestors identifying their offspring to find our parent. 

179 

180 DIEs are stored as a flattened tree. The top DIE is the ancestor 

181 of all DIEs in the unit. Each parent is guaranteed to be at 

182 an offset less than their children. In each generation of children 

183 the sibling with the closest offset not greater than our offset is 

184 our ancestor. 

185 """ 

186 # This code is called when get_parent notices that the _parent has 

187 # not been identified. To avoid execution for each sibling record all 

188 # the children of any parent iterated. Assuming get_parent will also be 

189 # called for siblings, it is more efficient if siblings references are 

190 # provided and no worse than a single walk if they are missing, while 

191 # stopping iteration early could result in O(n^2) walks. 

192 search = self.cu.get_top_DIE() 

193 while search.offset < self.offset: 

194 prev = search 

195 for child in search.iter_children(): 

196 child.set_parent(search) 

197 if child.offset <= self.offset: 

198 prev = child 

199 

200 # We also need to check the offset of the terminator DIE 

201 if search.has_children and search._terminator.offset <= self.offset: 

202 prev = search._terminator 

203 

204 # If we didn't find a closer parent, give up, don't loop. 

205 # Either we mis-parsed an ancestor or someone created a DIE 

206 # by an offset that was not actually the start of a DIE. 

207 if prev is search: 

208 raise ValueError("offset %s not in CU %s DIE tree" % 

209 (self.offset, self.cu.cu_offset)) 

210 

211 search = prev

223 def _parse_DIE(self): 

224 """ Parses the DIE info from the section, based on the abbreviation 

225 table of the CU 

226 """ 

227 structs = self.cu.structs 

228 

229 # A DIE begins with the abbreviation code. Read it and use it to 

230 # obtain the abbrev declaration for this DIE. 

231 # Note: here and elsewhere, preserve_stream_pos is used on operations 

232 # that manipulate the stream by reading data from it. 

233 self.abbrev_code = struct_parse( 

234 structs.Dwarf_uleb128(''), self.stream, self.offset) 

235 

236 # This may be a null entry 

237 if self.abbrev_code == 0: 

238 self.size = self.stream.tell() - self.offset 

239 return 

240 

241 abbrev_decl = self.cu.get_abbrev_table().get_abbrev(self.abbrev_code) 

242 self.tag = abbrev_decl['tag'] 

243 self.has_children = abbrev_decl.has_children() 

244 

245 # Guided by the attributes listed in the abbreviation declaration, parse 

246 # values from the stream. 

247 for spec in abbrev_decl['attr_spec']: 

248 form = spec.form 

249 name = spec.name 

250 attr_offset = self.stream.tell() 

251 indirection_length = 0 

252 # Special case here: the attribute value is stored in the attribute 

253 # definition in the abbreviation spec, not in the DIE itself. 

254 if form == 'DW_FORM_implicit_const': 

255 value = spec.value 

256 raw_value = value 

257 # Another special case: the attribute value is a form code followed by the real value in that form 

258 elif form == 'DW_FORM_indirect': 

259 (form, raw_value, indirection_length) = self._resolve_indirect() 

260 value = self._translate_attr_value(form, raw_value) 

261 else: 

262 raw_value = struct_parse(structs.Dwarf_dw_form[form], self.stream) 

263 value = self._translate_attr_value(form, raw_value) 

264 self.attributes[name] = AttributeValue( 

265 name=name, 

266 form=form, 

267 value=value, 

268 raw_value=raw_value, 

269 offset=attr_offset, 

270 indirection_length = indirection_length) 

271 

272 self.size = self.stream.tell() - self.offset

274 def _resolve_indirect(self): 

275 # Supports arbitrary indirection nesting (the standard doesn't prohibit that). 

276 # Expects the stream to be at the real form. 

277 # Returns (form, raw_value, length). 

278 structs = self.cu.structs 

279 length = 1 

280 real_form_code = struct_parse(structs.Dwarf_uleb128(''), self.stream) # Numeric form code 

281 while True: 

282 try: 

283 real_form = DW_FORM_raw2name[real_form_code] # Form name or exception if bogus code 

284 except KeyError as err: 

285 raise DWARFError('Found DW_FORM_indirect with unknown real form 0x%x' % real_form_code) 

286 

287 raw_value = struct_parse(structs.Dwarf_dw_form[real_form], self.stream) 

288 

289 if real_form != 'DW_FORM_indirect': # Happy path: one level of indirection 

290 return (real_form, raw_value, length) 

291 else: # Indirection cascade 

292 length += 1 

293 real_form_code = raw_value 

294 # And continue parsing 

295 # No explicit infinite loop guard because the stream will end eventually

297 def _translate_attr_value(self, form, raw_value): 

298 """ Translate a raw attr value according to the form 

299 """ 

300 # Indirect forms can only be parsed if the top DIE of this CU has already been parsed 

301 # and listed in the CU, since the top DIE would have to contain the DW_AT_xxx_base attributes. 

302 # This breaks if there is an indirect encoding in the top DIE itself before the 

303 # corresponding _base, and it was seen in the wild. 

304 # There is a hook in get_top_DIE() to resolve those lazily. 

305 translate_indirect = self.cu.has_top_DIE() or self.offset != self.cu.cu_die_offset 

306 value = None 

307 if form == 'DW_FORM_strp': 

308 with preserve_stream_pos(self.stream): 

309 value = self.dwarfinfo.get_string_from_table(raw_value) 

310 elif form == 'DW_FORM_line_strp': 

311 with preserve_stream_pos(self.stream): 

312 value = self.dwarfinfo.get_string_from_linetable(raw_value) 

313 elif form in ('DW_FORM_GNU_strp_alt', 'DW_FORM_strp_sup'): 

314 if self.dwarfinfo.supplementary_dwarfinfo: 

315 return self.dwarfinfo.supplementary_dwarfinfo.get_string_from_table(raw_value) 

316 else: 

317 value = raw_value 

318 elif form == 'DW_FORM_flag': 

319 value = not raw_value == 0 

320 elif form == 'DW_FORM_flag_present': 

321 value = True 

322 elif form in ('DW_FORM_addrx', 'DW_FORM_addrx1', 'DW_FORM_addrx2', 'DW_FORM_addrx3', 'DW_FORM_addrx4') and translate_indirect: 

323 value = self.cu.dwarfinfo.get_addr(self.cu, raw_value) 

324 elif form in ('DW_FORM_strx', 'DW_FORM_strx1', 'DW_FORM_strx2', 'DW_FORM_strx3', 'DW_FORM_strx4') and translate_indirect: 

325 stream = self.dwarfinfo.debug_str_offsets_sec.stream 

326 base_offset = _get_base_offset(self.cu, 'DW_AT_str_offsets_base') 

327 offset_size = 4 if self.cu.structs.dwarf_format == 32 else 8 

328 with preserve_stream_pos(stream): 

329 str_offset = struct_parse(self.cu.structs.Dwarf_offset(''), stream, base_offset + raw_value*offset_size) 

330 value = self.dwarfinfo.get_string_from_table(str_offset) 

331 elif form == 'DW_FORM_loclistx' and translate_indirect: 

332 value = _resolve_via_offset_table(self.dwarfinfo.debug_loclists_sec.stream, self.cu, raw_value, 'DW_AT_loclists_base') 

333 elif form == 'DW_FORM_rnglistx' and translate_indirect: 

334 value = _resolve_via_offset_table(self.dwarfinfo.debug_rnglists_sec.stream, self.cu, raw_value, 'DW_AT_rnglists_base') 

335 else: 

336 value = raw_value 

337 return value

339 def _translate_indirect_attributes(self): 

340 """ This is a hook to translate the DW_FORM_...x values in the top DIE 

341 once the top DIE is parsed to the end. They can't be translated  

342 while the top DIE is being parsed, because they implicitly make a 

343 reference to the DW_AT_xxx_base attribute in the same DIE that may 

344 not have been parsed yet. 

345 """ 

346 for key in self.attributes: 

347 attr = self.attributes[key] 

348 if attr.form in ('DW_FORM_strx', 'DW_FORM_strx1', 'DW_FORM_strx2', 'DW_FORM_strx3', 'DW_FORM_strx4', 

349 'DW_FORM_addrx', 'DW_FORM_addrx1', 'DW_FORM_addrx2', 'DW_FORM_addrx3', 'DW_FORM_addrx4', 

350 'DW_FORM_loclistx', 'DW_FORM_rnglistx'): 

351 # Can't change value in place, got to replace the whole attribute record 

352 self.attributes[key] = AttributeValue( 

353 name=attr.name, 

354 form=attr.form, 

355 value=self._translate_attr_value(attr.form, attr.raw_value), 

356 raw_value=attr.raw_value, 

357 offset=attr.offset, 

358 indirection_length=attr.indirection_length)
