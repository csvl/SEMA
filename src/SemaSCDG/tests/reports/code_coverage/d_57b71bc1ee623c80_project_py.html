241 def _register_object(self, obj, sim_proc_arch): 

242 """ 

243 This scans through an objects imports and hooks them with simprocedures from our library whenever possible 

244 """ 

245 

246 # Step 1: get the set of libraries we are allowed to use to resolve unresolved symbols 

247 missing_libs = [] 

248 for lib_name in self.loader.missing_dependencies: 

249 try: 

250 missing_libs.append(SIM_LIBRARIES[lib_name]) 

251 except KeyError: 

252 l.info("There are no simprocedures for missing library %s :(", lib_name) 

253 # additionally provide libraries we _have_ loaded as a fallback fallback 

254 # this helps in the case that e.g. CLE picked up a linux arm libc to satisfy an android arm binary 

255 for lib in self.loader.all_objects: 

256 if lib.provides in SIM_LIBRARIES: 

257 simlib = SIM_LIBRARIES[lib.provides] 

258 if simlib not in missing_libs: 

259 missing_libs.append(simlib) 

260 

261 # Step 2: Categorize every "import" symbol in each object. 

262 # If it's IGNORED, mark it for stubbing 

263 # If it's blacklisted, don't process it 

264 # If it matches a simprocedure we have, replace it 

265 for reloc in obj.imports.values(): 

266 # Step 2.1: Quick filter on symbols we really don't care about 

267 func = reloc.symbol 

268 if func is None: 

269 continue 

270 if not func.is_function and func.type != cle.backends.symbol.SymbolType.TYPE_NONE: 

271 continue 

272 if func.resolvedby is None: 

273 # I don't understand the binary which made me add this case. If you are debugging and see this comment, 

274 # good luck. 

275 # ref: https://github.com/angr/angr/issues/1782 

276 # (I also don't know why the TYPE_NONE check in the previous clause is there but I can't find a ref for 

277 # that. they are probably related.) 

278 # (I believe the TYPE_NONE check is to support ELF object files) 

279 continue 

280 if not reloc.resolved: 

281 # This is a hack, effectively to support Binary Ninja, which doesn't provide access to dependency 

282 # library names. The backend creates the Relocation objects, but leaves them unresolved so that 

283 # we can try to guess them here. 

284 if reloc.owner.guess_simprocs: 

285 l.debug("Looking for matching SimProcedure for unresolved %s from %s with hint %s", 

286 func.name, reloc.owner, reloc.owner.guess_simprocs_hint) 

287 self._guess_simprocedure(func, reloc.owner.guess_simprocs_hint) 

288 else: 

289 l.debug("Ignoring unresolved import '%s' from %s ...?", func.name, reloc.owner) 

290 continue 

291 export = reloc.resolvedby 

292 if self.is_hooked(export.rebased_addr): 

293 l.debug("Already hooked %s (%s)", export.name, export.owner) 

294 continue 

295 

296 # Step 2.2: If this function has been resolved by a static dependency, 

297 # check if we actually can and want to replace it with a SimProcedure. 

298 # We opt out of this step if it is blacklisted by ignore_functions, which 

299 # will cause it to be replaced by ReturnUnconstrained later. 

300 if export.owner is not self.loader._extern_object and \ 

301 export.name not in self._ignore_functions: 

302 if self._check_user_blacklists(export.name): 

303 continue 

304 owner_name = export.owner.provides 

305 if isinstance(self.loader.main_object, cle.backends.pe.PE): 

306 owner_name = owner_name.lower() 

307 if owner_name not in SIM_LIBRARIES: 

308 continue 

309 sim_lib = SIM_LIBRARIES[owner_name] 

310 if not sim_lib.has_implementation(export.name): 

311 continue 

312 l.info("Using builtin SimProcedure for %s from %s", export.name, sim_lib.name) 

313 self.hook_symbol(export.rebased_addr, sim_lib.get(export.name, sim_proc_arch)) 

314 

315 # Step 2.3: If 2.2 didn't work, check if the symbol wants to be resolved 

316 # by a library we already know something about. Resolve it appropriately. 

317 # Note that _check_user_blacklists also includes _ignore_functions. 

318 # An important consideration is that even if we're stubbing a function out, 

319 # we still want to try as hard as we can to figure out where it comes from 

320 # so we can get the calling convention as close to right as possible. 

321 elif reloc.resolvewith is not None and reloc.resolvewith in SIM_LIBRARIES: 

322 sim_lib = SIM_LIBRARIES[reloc.resolvewith] 

323 if self._check_user_blacklists(export.name): 

324 if not func.is_weak: 

325 l.info("Using stub SimProcedure for unresolved %s from %s", func.name, sim_lib.name) 

326 self.hook_symbol(export.rebased_addr, sim_lib.get_stub(export.name, sim_proc_arch)) 

327 else: 

328 l.info("Using builtin SimProcedure for unresolved %s from %s", export.name, sim_lib.name) 

329 self.hook_symbol(export.rebased_addr, sim_lib.get(export.name, sim_proc_arch)) 

330 

331 # Step 2.4: If 2.3 didn't work (the symbol didn't request a provider we know of), try 

332 # looking through each of the SimLibraries we're using to resolve unresolved 

333 # functions. If any of them know anything specifically about this function, 

334 # resolve it with that. As a final fallback, just ask any old SimLibrary 

335 # to resolve it. 

336 elif missing_libs: 

337 for sim_lib in missing_libs: 

338 if sim_lib.has_metadata(export.name): 

339 if self._check_user_blacklists(export.name): 

340 if not func.is_weak: 

341 l.info("Using stub SimProcedure for unresolved %s from %s", export.name, sim_lib.name) 

342 self.hook_symbol(export.rebased_addr, sim_lib.get_stub(export.name, sim_proc_arch)) 

343 else: 

344 l.info("Using builtin SimProcedure for unresolved %s from %s", export.name, sim_lib.name) 

345 self.hook_symbol(export.rebased_addr, sim_lib.get(export.name, sim_proc_arch)) 

346 break 

347 else: 

348 if not func.is_weak: 

349 l.info("Using stub SimProcedure for unresolved %s", export.name) 

350 the_lib = missing_libs[0] 

351 if export.name and export.name.startswith("_Z"): 

352 # GNU C++ name. Use a C++ library to create the stub 

353 if 'libstdc++.so' in SIM_LIBRARIES: 

354 the_lib = SIM_LIBRARIES['libstdc++.so'] 

355 else: 

356 l.critical("Does not find any C++ library in SIM_LIBRARIES. We may not correctly " 

357 "create the stub or resolve the function prototype for name %s.", export.name) 

358 

359 self.hook_symbol(export.rebased_addr, the_lib.get(export.name, sim_proc_arch)) 

360 

361 # Step 2.5: If the requesting object wants us to guess simprocedures, do the guessing 

362 elif reloc.owner.guess_simprocs and self._guess_simprocedure(func, reloc.owner.guess_simprocs_hint): 

363 continue 

364 

365 # Step 2.6: If 2.4/2.5 didn't work (we have NO SimLibraries to work with), just 

366 # use the vanilla ReturnUnconstrained, assuming that this isn't a weak func 

367 elif not func.is_weak: 

368 l.info("Using stub SimProcedure for unresolved %s", export.name) 

369 self.hook_symbol(export.rebased_addr, SIM_PROCEDURES['stubs']['ReturnUnconstrained'](display_name=export.name, is_stub=True))

372 """

418 def hook(self, addr, hook=None, length=0, kwargs=None, replace=False): 

419 """ 

420 Hook a section of code with a custom function. This is used internally to provide symbolic 

421 summaries of library functions, and can be used to instrument execution or to modify 

422 control flow. 

423 

424 When hook is not specified, it returns a function decorator that allows easy hooking. 

425 Usage:: 

426 

427 # Assuming proj is an instance of angr.Project, we will add a custom hook at the entry 

428 # point of the project. 

429 @proj.hook(proj.entry) 

430 def my_hook(state): 

431 print("Welcome to execution!") 

432 

433 :param addr: The address to hook. 

434 :param hook: A :class:`angr.project.Hook` describing a procedure to run at the 

435 given address. You may also pass in a SimProcedure class or a function 

436 directly and it will be wrapped in a Hook object for you. 

437 :param length: If you provide a function for the hook, this is the number of bytes 

438 that will be skipped by executing the hook by default. 

439 :param kwargs: If you provide a SimProcedure for the hook, these are the keyword 

440 arguments that will be passed to the procedure's `run` method 

441 eventually. 

442 :param replace: Control the behavior on finding that the address is already hooked. If 

443 true, silently replace the hook. If false (default), warn and do not 

444 replace the hook. If none, warn and replace the hook. 

445 """ 

446 if hook is None: 

447 # if we haven't been passed a thing to hook with, assume we're being used as a decorator 

448 return self._hook_decorator(addr, length=length, kwargs=kwargs) 

449 

450 if kwargs is None: kwargs = {} 

451 

452 l.debug('hooking %s with %s', self._addr_to_str(addr), str(hook)) 

453 

454 if self.is_hooked(addr): 

455 if replace is True: 

456 pass 

457 elif replace is False: 

458 l.warning("Address is already hooked, during hook(%s, %s). Not re-hooking.", self._addr_to_str(addr), hook) 

459 return 

460 else: 

461 l.warning("Address is already hooked, during hook(%s, %s). Re-hooking.", self._addr_to_str(addr), hook) 

462 

463 if isinstance(hook, type): 

464 raise TypeError("Please instanciate your SimProcedure before hooking with it") 

465 

466 if callable(hook): 

467 hook = SIM_PROCEDURES['stubs']['UserHook'](user_func=hook, length=length, **kwargs) 

468 

469 self._sim_procedures[addr] = hook

507 def hook_symbol(self, symbol_name, simproc, kwargs=None, replace=None): 

508 """ 

509 Resolve a dependency in a binary. Looks up the address of the given symbol, and then hooks that 

510 address. If the symbol was not available in the loaded libraries, this address may be provided 

511 by the CLE externs object. 

512 

513 Additionally, if instead of a symbol name you provide an address, some secret functionality will 

514 kick in and you will probably just hook that address, UNLESS you're on powerpc64 ABIv1 or some 

515 yet-unknown scary ABI that has its function pointers point to something other than the actual 

516 functions, in which case it'll do the right thing. 

517 

518 :param symbol_name: The name of the dependency to resolve. 

519 :param simproc: The SimProcedure instance (or function) with which to hook the symbol 

520 :param kwargs: If you provide a SimProcedure for the hook, these are the keyword 

521 arguments that will be passed to the procedure's `run` method 

522 eventually. 

523 :param replace: Control the behavior on finding that the address is already hooked. If 

524 true, silently replace the hook. If false, warn and do not replace the 

525 hook. If none (default), warn and replace the hook. 

526 :returns: The address of the new symbol. 

527 :rtype: int 

528 """ 

529 if type(symbol_name) is not int: 

530 sym = self.loader.find_symbol(symbol_name) 

531 if sym is None: 

532 # it could be a previously unresolved weak symbol..? 

533 new_sym = None 

534 for reloc in self.loader.find_relevant_relocations(symbol_name): 

535 if not reloc.symbol.is_weak: 

536 raise Exception("Symbol is strong but we couldn't find its resolution? Report to @rhelmot.") 

537 if new_sym is None: 

538 new_sym = self.loader.extern_object.make_extern(symbol_name) 

539 reloc.resolve(new_sym) 

540 reloc.relocate([]) 

541 

542 if new_sym is None: 

543 l.error("Could not find symbol %s", symbol_name) 

544 return None 

545 sym = new_sym 

546 

547 basic_addr = sym.rebased_addr 

548 else: 

549 basic_addr = symbol_name 

550 symbol_name = None 

551 

552 hook_addr, _ = self.simos.prepare_function_symbol(symbol_name, basic_addr=basic_addr) 

553 

554 self.hook(hook_addr, simproc, kwargs=kwargs, replace=replace) 

555 return hook_addr

557 def is_symbol_hooked(self, symbol_name): 

558 """ 

559 Check if a symbol is already hooked. 

560 

561 :param str symbol_name: Name of the symbol. 

562 :return: True if the symbol can be resolved and is hooked, False otherwise. 

563 :rtype: bool 

564 """ 

565 sym = self.loader.find_symbol(symbol_name) 

566 if sym is None: 

567 l.warning("Could not find symbol %s", symbol_name) 

568 return False 

569 hook_addr, _ = self.simos.prepare_function_symbol(symbol_name, basic_addr=sym.rebased_addr) 

570 return self.is_hooked(hook_addr) 

571 

572 def unhook_symbol(self, symbol_name): 

573 """ 

574 Remove the hook on a symbol. 

575 This function will fail if the symbol is provided by the extern object, as that would result in a state where 

576 analysis would be unable to cope with a call to this symbol. 

577 """ 

578 sym = self.loader.find_symbol(symbol_name) 

579 if sym is None: 

580 l.warning("Could not find symbol %s", symbol_name) 

581 return False 

582 if sym.owner is self.loader._extern_object: 

583 l.warning("Refusing to unhook external symbol %s, replace it with another hook if you want to change it", 

584 symbol_name) 

585 return False 

586 

587 hook_addr, _ = self.simos.prepare_function_symbol(symbol_name, basic_addr=sym.rebased_addr) 

588 self.unhook(hook_addr) 

589 return True

591 def rehook_symbol(self, new_address, symbol_name, stubs_on_sync): 

592 """ 

593 Move the hook for a symbol to a specific address 

594 :param new_address: the new address that will trigger the SimProc execution 

595 :param symbol_name: the name of the symbol (f.i. strcmp ) 

596 :return: None 

597 """ 

598 new_sim_procedures = {} 

599 for key_address, simproc_obj in self._sim_procedures.items(): 

600 

601 # if we don't want stubs during the sync let's skip those, we will execute the real function. 

602 if not stubs_on_sync and simproc_obj.is_stub: 

603 continue 

604 

605 if simproc_obj.display_name == symbol_name: 

606 new_sim_procedures[new_address] = simproc_obj 

607 else: 

608 new_sim_procedures[key_address] = simproc_obj 

609 

610 self._sim_procedures = new_sim_procedures

614 # 

615 

616 def execute(self, *args, **kwargs): 

617 """ 

618 This function is a symbolic execution helper in the simple style 

619 supported by triton and manticore. It designed to be run after 

620 setting up hooks (see Project.hook), in which the symbolic state 

621 can be checked. 

622 

623 This function can be run in three different ways: 

624 

625 - When run with no parameters, this function begins symbolic execution 

626 from the entrypoint. 

627 - It can also be run with a "state" parameter specifying a SimState to 

628 begin symbolic execution from. 

629 - Finally, it can accept any arbitrary keyword arguments, which are all 

630 passed to project.factory.full_init_state. 

631 

632 If symbolic execution finishes, this function returns the resulting 

633 simulation manager. 

634 """ 

635 

636 if args: 

637 state = args[0] 

638 else: 

639 state = self.factory.full_init_state(**kwargs) 

640 

641 pg = self.factory.simulation_manager(state) 

642 self._executing = True 

643 return pg.run(until=lambda lpg: not self._executing) 

644 

645 def terminate_execution(self): 

646 """ 

647 Terminates a symbolic execution that was started with Project.execute(). 

648 """ 

649 self._executing = False

692 def _store(self, container): 

693 # If container is a filename. 

694 if isinstance(container, str): 

695 with open(container, 'wb') as f: 

696 try: 

697 pickle.dump(self, f, pickle.HIGHEST_PROTOCOL) 

698 except RuntimeError as e: # maximum recursion depth can be reached here 

699 l.error("Unable to store Project: '%s' during pickling", e) 

700 

701 # If container is an open file. 

702 elif isinstance(container, IOBase): 

703 try: 

704 pickle.dump(self, container, pickle.HIGHEST_PROTOCOL) 

705 except RuntimeError as e: # maximum recursion depth can be reached here 

706 l.error("Unable to store Project: '%s' during pickling", e) 

707 

708 # If container is just a variable. 

709 else: 

710 try: 

711 container = pickle.dumps(self, pickle.HIGHEST_PROTOCOL) 

712 except RuntimeError as e: # maximum recursion depth can be reached here 

713 l.error("Unable to store Project: '%s' during pickling", e)

715 @staticmethod 

716 def _load(container): 

717 if isinstance(container, str): 

718 # If container is a filename. 

719 if all(c in string.printable for c in container) and os.path.exists(container): 

720 with open(container, 'rb') as f: 

721 return pickle.load(f) 

722 

723 # If container is a pickle string. 

724 else: 

725 return pickle.loads(container) 

726 

727 # If container is an open file 

728 elif isinstance(container, IOBase): 

729 return pickle.load(container) 

730 

731 # What else could it be? 

732 else: 

733 l.error("Cannot unpickle container of type %s", type(container)) 

734 return None
