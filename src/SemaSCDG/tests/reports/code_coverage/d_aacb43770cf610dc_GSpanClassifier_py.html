348 def get_stat_classifier(self,target='classification'): 

349 if target == 'classification' and not self.predictions: 

350 self.log.info('Need to classify first\n') 

351 return 

352 elif target == 'detection' and not self.predictions_clean: 

353 self.log.info('Need to classify cleanwares first\n') 

354 return 

355 elif target not in ['classification','detection']: 

356 self.log.info('Not valid target\n') 

357 return 

358 else : 

359 pass 

360 

361 dico ={self.families[i]: i for i in range(len(self.families))} 

362 sample_per_family = [0 for i in range(len(self.families))] 

363 tp = [0 for i in range(len(self.families))] 

364 fp = [0 for i in range(len(self.families))] 

365 conf_matrix = [[0 for i in range(len(self.families))] for i in range(len(self.families))] 

366 

367 for i in range(len(self.predictions)): 

368 tab = self.predictions[i] 

369 sample_per_family[dico[tab[1]]] += 1 

370 if tab[0] == tab[1]: 

371 tp[dico[tab[1]]] += 1 

372 else : 

373 fp[dico[tab[0]]] += 1 

374 conf_matrix[dico[tab[1]]][dico[tab[0]]] += 1 

375 

376 total_sample = sum(sample_per_family) 

377 precision = [] 

378 recall = [] 

379 for i in range(len(self.families)): 

380 if (tp[i]+fp[i]) != 0: 

381 precision = precision + [(tp[i]/(tp[i]+fp[i]))*sample_per_family[i]/total_sample] 

382 else : 

383 precision = precision + [0] 

384 if sample_per_family[i] !=0: 

385 recall = recall + [(tp[i]/sample_per_family[i])*sample_per_family[i]/total_sample] 

386 else : 

387 recall = recall + [0] 

388 precision = sum(precision) 

389 recall = sum(recall) 

390 fscore = (2* precision * recall)/(precision + recall) 

391 logging.basicConfig(level=logging.INFO) 

392 self.log.info("Precision obtained : "+str(precision)) 

393 self.log.info("Recall obtained : "+str(recall)) 

394 self.log.info("Fscore obtained : "+str(fscore)) 

395 

396 self.fscore = fscore
