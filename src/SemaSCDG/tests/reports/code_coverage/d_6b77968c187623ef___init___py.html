279class Process(object): # noqa: UP004 

280 """Represents an OS process with the given PID. 

281 If PID is omitted current process PID (os.getpid()) is used. 

282 Raise NoSuchProcess if PID does not exist. 

283 

284 Note that most of the methods of this class do not make sure 

285 the PID of the process being queried has been reused over time. 

286 That means you might end up retrieving an information referring 

287 to another process in case the original one this instance 

288 refers to is gone in the meantime. 

289 

290 The only exceptions for which process identity is pre-emptively 

291 checked and guaranteed are: 

292 

293 - parent() 

294 - children() 

295 - nice() (set) 

296 - ionice() (set) 

297 - rlimit() (set) 

298 - cpu_affinity (set) 

299 - suspend() 

300 - resume() 

301 - send_signal() 

302 - terminate() 

303 - kill() 

304 

305 To prevent this problem for all other methods you can: 

306 - use is_running() before querying the process 

307 - if you're continuously iterating over a set of Process 

308 instances use process_iter() which pre-emptively checks 

309 process identity for every yielded instance 

310 """ 

311 

312 def __init__(self, pid=None): 

313 self._init(pid)

315 def _init(self, pid, _ignore_nsp=False): 

316 if pid is None: 

317 pid = os.getpid() 

318 else: 

319 if not _PY3 and not isinstance(pid, (int, long)): 

320 raise TypeError('pid must be an integer (got %r)' % pid) 

321 if pid < 0: 

322 raise ValueError('pid must be a positive integer (got %s)' 

323 % pid) 

324 try: 

325 _psplatform.cext.check_pid_range(pid) 

326 except OverflowError: 

327 raise NoSuchProcess( 

328 pid, 

329 msg='process PID out of range (got %s)' % pid, 

330 ) 

331 

332 self._pid = pid 

333 self._name = None 

334 self._exe = None 

335 self._create_time = None 

336 self._gone = False 

337 self._pid_reused = False 

338 self._hash = None 

339 self._lock = threading.RLock() 

340 # used for caching on Windows only (on POSIX ppid may change) 

341 self._ppid = None 

342 # platform-specific modules define an _psplatform.Process 

343 # implementation class 

344 self._proc = _psplatform.Process(pid) 

345 self._last_sys_cpu_times = None 

346 self._last_proc_cpu_times = None 

347 self._exitcode = _SENTINEL 

348 # cache creation time for later use in is_running() method 

349 try: 

350 self.create_time() 

351 except AccessDenied: 

352 # We should never get here as AFAIK we're able to get 

353 # process creation time on all platforms even as a 

354 # limited user. 

355 pass 

356 except ZombieProcess: 

357 # Zombies can still be queried by this class (although 

358 # not always) and pids() return them so just go on. 

359 pass 

360 except NoSuchProcess: 

361 if not _ignore_nsp: 

362 raise NoSuchProcess(pid, msg='process PID not found') 

363 else: 

364 self._gone = True 

365 # This pair is supposed to identify a Process instance 

366 # univocally over time (the PID alone is not enough as 

367 # it might refer to a process whose PID has been reused). 

368 # This will be used later in __eq__() and is_running(). 

369 self._ident = (self.pid, self._create_time)

371 def __str__(self): 

372 info = collections.OrderedDict() 

373 info["pid"] = self.pid 

374 if self._name: 

375 info['name'] = self._name 

376 with self.oneshot(): 

377 try: 

378 info["name"] = self.name() 

379 info["status"] = self.status() 

380 except ZombieProcess: 

381 info["status"] = "zombie" 

382 except NoSuchProcess: 

383 info["status"] = "terminated" 

384 except AccessDenied: 

385 pass 

386 if self._exitcode not in (_SENTINEL, None): 

387 info["exitcode"] = self._exitcode 

388 if self._create_time is not None: 

389 info['started'] = _pprint_secs(self._create_time) 

390 return "%s.%s(%s)" % ( 

391 self.__class__.__module__, 

392 self.__class__.__name__, 

393 ", ".join(["%s=%r" % (k, v) for k, v in info.items()]))

397 def __eq__(self, other): 

398 # Test for equality with another Process object based 

399 # on PID and creation time. 

400 if not isinstance(other, Process): 

401 return NotImplemented 

402 if OPENBSD or NETBSD: # pragma: no cover 

403 # Zombie processes on Open/NetBSD have a creation time of 

404 # 0.0. This covers the case when a process started normally 

405 # (so it has a ctime), then it turned into a zombie. It's 

406 # important to do this because is_running() depends on 

407 # __eq__. 

408 pid1, ctime1 = self._ident 

409 pid2, ctime2 = other._ident 

410 if pid1 == pid2: 

411 if ctime1 and not ctime2: 

412 try: 

413 return self.status() == STATUS_ZOMBIE 

414 except Error: 

415 pass 

416 return self._ident == other._ident 

417 

418 def __ne__(self, other): 

419 return not self == other 

420 

421 def __hash__(self): 

422 if self._hash is None: 

423 self._hash = hash(self._ident) 

424 return self._hash

426 def _raise_if_pid_reused(self): 

427 """Raises NoSuchProcess in case process PID has been reused.""" 

428 if not self.is_running() and self._pid_reused: 

429 # We may directly raise NSP in here already if PID is just 

430 # not running, but I prefer NSP to be raised naturally by 

431 # the actual Process API call. This way unit tests will tell 

432 # us if the API is broken (aka don't raise NSP when it 

433 # should). We also remain consistent with all other "get" 

434 # APIs which don't use _raise_if_pid_reused(). 

435 msg = "process no longer exists and its PID has been reused" 

436 raise NoSuchProcess(self.pid, self._name, msg=msg)

445 @contextlib.contextmanager 

446 def oneshot(self): 

447 """Utility context manager which considerably speeds up the 

448 retrieval of multiple process information at the same time. 

449 

450 Internally different process info (e.g. name, ppid, uids, 

451 gids, ...) may be fetched by using the same routine, but 

452 only one information is returned and the others are discarded. 

453 When using this context manager the internal routine is 

454 executed once (in the example below on name()) and the 

455 other info are cached. 

456 

457 The cache is cleared when exiting the context manager block. 

458 The advice is to use this every time you retrieve more than 

459 one information about the process. If you're lucky, you'll 

460 get a hell of a speedup. 

461 

462 >>> import psutil 

463 >>> p = psutil.Process() 

464 >>> with p.oneshot(): 

465 ... p.name() # collect multiple info 

466 ... p.cpu_times() # return cached value 

467 ... p.cpu_percent() # return cached value 

468 ... p.create_time() # return cached value 

469 ... 

470 >>> 

471 """ 

472 with self._lock: 

473 if hasattr(self, "_cache"): 

474 # NOOP: this covers the use case where the user enters the 

475 # context twice: 

476 # 

477 # >>> with p.oneshot(): 

478 # ... with p.oneshot(): 

479 # ... 

480 # 

481 # Also, since as_dict() internally uses oneshot() 

482 # I expect that the code below will be a pretty common 

483 # "mistake" that the user will make, so let's guard 

484 # against that: 

485 # 

486 # >>> with p.oneshot(): 

487 # ... p.as_dict() 

488 # ... 

489 yield 

490 else: 

491 try: 

492 # cached in case cpu_percent() is used 

493 self.cpu_times.cache_activate(self) 

494 # cached in case memory_percent() is used 

495 self.memory_info.cache_activate(self) 

496 # cached in case parent() is used 

497 self.ppid.cache_activate(self) 

498 # cached in case username() is used 

499 if POSIX: 

500 self.uids.cache_activate(self) 

501 # specific implementation cache 

502 self._proc.oneshot_enter() 

503 yield 

504 finally: 

505 self.cpu_times.cache_deactivate(self) 

506 self.memory_info.cache_deactivate(self) 

507 self.ppid.cache_deactivate(self) 

508 if POSIX: 

509 self.uids.cache_deactivate(self) 

510 self._proc.oneshot_exit()

512 def as_dict(self, attrs=None, ad_value=None): 

513 """Utility method returning process information as a 

514 hashable dictionary. 

515 If *attrs* is specified it must be a list of strings 

516 reflecting available Process class' attribute names 

517 (e.g. ['cpu_times', 'name']) else all public (read 

518 only) attributes are assumed. 

519 *ad_value* is the value which gets assigned in case 

520 AccessDenied or ZombieProcess exception is raised when 

521 retrieving that particular process information. 

522 """ 

523 valid_names = _as_dict_attrnames 

524 if attrs is not None: 

525 if not isinstance(attrs, (list, tuple, set, frozenset)): 

526 raise TypeError("invalid attrs type %s" % type(attrs)) 

527 attrs = set(attrs) 

528 invalid_names = attrs - valid_names 

529 if invalid_names: 

530 raise ValueError("invalid attr name%s %s" % ( 

531 "s" if len(invalid_names) > 1 else "", 

532 ", ".join(map(repr, invalid_names)))) 

533 

534 retdict = {} 

535 ls = attrs or valid_names 

536 with self.oneshot(): 

537 for name in ls: 

538 try: 

539 if name == 'pid': 

540 ret = self.pid 

541 else: 

542 meth = getattr(self, name) 

543 ret = meth() 

544 except (AccessDenied, ZombieProcess): 

545 ret = ad_value 

546 except NotImplementedError: 

547 # in case of not implemented functionality (may happen 

548 # on old or exotic systems) we want to crash only if 

549 # the user explicitly asked for that particular attr 

550 if attrs: 

551 raise 

552 continue 

553 retdict[name] = ret 

554 return retdict

556 def parent(self): 

557 """Return the parent process as a Process object pre-emptively 

558 checking whether PID has been reused. 

559 If no parent is known return None. 

560 """ 

561 lowest_pid = _LOWEST_PID if _LOWEST_PID is not None else pids()[0] 

562 if self.pid == lowest_pid: 

563 return None 

564 ppid = self.ppid() 

565 if ppid is not None: 

566 ctime = self.create_time() 

567 try: 

568 parent = Process(ppid) 

569 if parent.create_time() <= ctime: 

570 return parent 

571 # ...else ppid has been reused by another process 

572 except NoSuchProcess: 

573 pass

575 def parents(self): 

576 """Return the parents of this process as a list of Process 

577 instances. If no parents are known return an empty list. 

578 """ 

579 parents = [] 

580 proc = self.parent() 

581 while proc is not None: 

582 parents.append(proc) 

583 proc = proc.parent() 

584 return parents

586 def is_running(self): 

587 """Return whether this process is running. 

588 It also checks if PID has been reused by another process in 

589 which case return False. 

590 """ 

591 if self._gone or self._pid_reused: 

592 return False 

593 try: 

594 # Checking if PID is alive is not enough as the PID might 

595 # have been reused by another process: we also want to 

596 # verify process identity. 

597 # Process identity / uniqueness over time is guaranteed by 

598 # (PID + creation time) and that is verified in __eq__. 

599 self._pid_reused = self != Process(self.pid) 

600 return not self._pid_reused 

601 except ZombieProcess: 

602 # We should never get here as it's already handled in 

603 # Process.__init__; here just for extra safety. 

604 return True 

605 except NoSuchProcess: 

606 self._gone = True 

607 return False

612 def ppid(self): 

613 """The process parent PID. 

614 On Windows the return value is cached after first call. 

615 """ 

616 # On POSIX we don't want to cache the ppid as it may unexpectedly 

617 # change to 1 (init) in case this process turns into a zombie: 

618 # https://github.com/giampaolo/psutil/issues/321 

619 # http://stackoverflow.com/questions/356722/ 

620 

621 # XXX should we check creation time here rather than in 

622 # Process.parent()? 

623 self._raise_if_pid_reused() 

624 if POSIX: 

625 return self._proc.ppid() 

626 else: # pragma: no cover 

627 self._ppid = self._ppid or self._proc.ppid() 

628 return self._ppid

630 def name(self): 

631 """The process name. The return value is cached after first call.""" 

632 # Process name is only cached on Windows as on POSIX it may 

633 # change, see: 

634 # https://github.com/giampaolo/psutil/issues/692 

635 if WINDOWS and self._name is not None: 

636 return self._name 

637 name = self._proc.name() 

638 if POSIX and len(name) >= 15: 

639 # On UNIX the name gets truncated to the first 15 characters. 

640 # If it matches the first part of the cmdline we return that 

641 # one instead because it's usually more explicative.

661 def exe(self): 

662 """The process executable as an absolute path. 

663 May also be an empty string. 

664 The return value is cached after first call. 

665 """ 

666 def guess_it(fallback): 

667 # try to guess exe from cmdline[0] in absence of a native 

668 # exe representation 

669 cmdline = self.cmdline() 

670 if cmdline and hasattr(os, 'access') and hasattr(os, 'X_OK'): 

671 exe = cmdline[0] # the possible exe 

672 # Attempt to guess only in case of an absolute path. 

673 # It is not safe otherwise as the process might have 

674 # changed cwd. 

675 if (os.path.isabs(exe) and 

676 os.path.isfile(exe) and 

677 os.access(exe, os.X_OK)): 

678 return exe 

679 if isinstance(fallback, AccessDenied): 

680 raise fallback 

681 return fallback 

682 

683 if self._exe is None: 

684 try: 

685 exe = self._proc.exe()

691 # empty string; if that's the case we don't want to 

692 # raise AD while guessing from the cmdline

901 def children(self, recursive=False): 

902 """Return the children of this process as a list of Process 

903 instances, pre-emptively checking whether PID has been reused. 

904 If *recursive* is True return all the parent descendants. 

905 

906 Example (A == this process): 

907 

908 A ─┐ 

909 │ 

910 ├─ B (child) ─┐ 

911 │ └─ X (grandchild) ─┐ 

912 │ └─ Y (great grandchild) 

913 ├─ C (child) 

914 └─ D (child) 

915 

916 >>> import psutil 

917 >>> p = psutil.Process() 

918 >>> p.children() 

919 B, C, D 

920 >>> p.children(recursive=True) 

921 B, X, Y, C, D 

922 

923 Note that in the example above if process X disappears 

924 process Y won't be listed as the reference to process A 

925 is lost. 

926 """ 

927 self._raise_if_pid_reused() 

928 ppid_map = _ppid_map() 

929 ret = [] 

930 if not recursive: 

931 for pid, ppid in ppid_map.items(): 

932 if ppid == self.pid: 

933 try: 

934 child = Process(pid) 

935 # if child happens to be older than its parent 

936 # (self) it means child's PID has been reused 

937 if self.create_time() <= child.create_time(): 

938 ret.append(child) 

939 except (NoSuchProcess, ZombieProcess): 

940 pass 

941 else: 

942 # Construct a {pid: [child pids]} dict 

943 reverse_ppid_map = collections.defaultdict(list) 

944 for pid, ppid in ppid_map.items(): 

945 reverse_ppid_map[ppid].append(pid) 

946 # Recursively traverse that dict, starting from self.pid, 

947 # such that we only call Process() on actual children 

948 seen = set() 

949 stack = [self.pid] 

950 while stack: 

951 pid = stack.pop() 

952 if pid in seen: 

953 # Since pids can be reused while the ppid_map is 

954 # constructed, there may be rare instances where 

955 # there's a cycle in the recorded process "tree". 

956 continue 

957 seen.add(pid) 

958 for child_pid in reverse_ppid_map[pid]: 

959 try: 

960 child = Process(child_pid) 

961 # if child happens to be older than its parent 

962 # (self) it means child's PID has been reused 

963 intime = self.create_time() <= child.create_time() 

964 if intime: 

965 ret.append(child) 

966 stack.append(child_pid) 

967 except (NoSuchProcess, ZombieProcess): 

968 pass 

969 return ret

971 def cpu_percent(self, interval=None): 

972 """Return a float representing the current process CPU 

973 utilization as a percentage. 

974 

975 When *interval* is 0.0 or None (default) compares process times 

976 to system CPU times elapsed since last call, returning 

977 immediately (non-blocking). That means that the first time 

978 this is called it will return a meaningful 0.0 value. 

979 

980 When *interval* is > 0.0 compares process times to system CPU 

981 times elapsed before and after the interval (blocking). 

982 

983 In this case is recommended for accuracy that this function 

984 be called with at least 0.1 seconds between calls. 

985 

986 A value > 100.0 can be returned in case of processes running 

987 multiple threads on different CPU cores. 

988 

989 The returned value is explicitly NOT split evenly between 

990 all available logical CPUs. This means that a busy loop process 

991 running on a system with 2 logical CPUs will be reported as 

992 having 100% CPU utilization instead of 50%. 

993 

994 Examples: 

995 

996 >>> import psutil 

997 >>> p = psutil.Process(os.getpid()) 

998 >>> # blocking 

999 >>> p.cpu_percent(interval=1) 

1000 2.0 

1001 >>> # non-blocking (percentage since last call) 

1002 >>> p.cpu_percent(interval=None) 

1003 2.9 

1004 >>> 

1005 """ 

1006 blocking = interval is not None and interval > 0.0 

1007 if interval is not None and interval < 0: 

1008 raise ValueError("interval is not positive (got %r)" % interval) 

1009 num_cpus = cpu_count() or 1 

1010 

1011 def timer(): 

1012 return _timer() * num_cpus 

1013 

1014 if blocking: 

1015 st1 = timer() 

1016 pt1 = self._proc.cpu_times() 

1017 time.sleep(interval) 

1018 st2 = timer() 

1019 pt2 = self._proc.cpu_times() 

1020 else: 

1021 st1 = self._last_sys_cpu_times 

1022 pt1 = self._last_proc_cpu_times 

1023 st2 = timer() 

1024 pt2 = self._proc.cpu_times() 

1025 if st1 is None or pt1 is None: 

1026 self._last_sys_cpu_times = st2 

1027 self._last_proc_cpu_times = pt2 

1028 return 0.0 

1029 

1030 delta_proc = (pt2.user - pt1.user) + (pt2.system - pt1.system) 

1031 delta_time = st2 - st1 

1032 # reset values for next call in case of interval == None 

1033 self._last_sys_cpu_times = st2 

1034 self._last_proc_cpu_times = pt2 

1035 

1036 try: 

1037 # This is the utilization split evenly between all CPUs. 

1038 # E.g. a busy loop process on a 2-CPU-cores system at this 

1039 # point is reported as 50% instead of 100%. 

1040 overall_cpus_percent = ((delta_proc / delta_time) * 100) 

1041 except ZeroDivisionError: 

1042 # interval was too low 

1043 return 0.0 

1044 else: 

1045 # Note 1: 

1046 # in order to emulate "top" we multiply the value for the num 

1047 # of CPU cores. This way the busy process will be reported as 

1048 # having 100% (or more) usage. 

1049 # 

1050 # Note 2: 

1051 # taskmgr.exe on Windows differs in that it will show 50% 

1052 # instead. 

1053 # 

1054 # Note 3: 

1055 # a percentage > 100 is legitimate as it can result from a 

1056 # process with multiple threads running on different CPU 

1057 # cores (top does the same), see: 

1058 # http://stackoverflow.com/questions/1032357 

1059 # https://github.com/giampaolo/psutil/issues/474 

1060 single_cpu_percent = overall_cpus_percent * num_cpus 

1061 return round(single_cpu_percent, 1)

1311class Popen(Process): 

1312 """Same as subprocess.Popen, but in addition it provides all 

1313 psutil.Process methods in a single class. 

1314 For the following methods which are common to both classes, psutil 

1315 implementation takes precedence: 

1316 

1317 * send_signal() 

1318 * terminate() 

1319 * kill() 

1320 

1321 This is done in order to avoid killing another process in case its 

1322 PID has been reused, fixing BPO-6973. 

1323 

1324 >>> import psutil 

1325 >>> from subprocess import PIPE 

1326 >>> p = psutil.Popen(["python", "-c", "print 'hi'"], stdout=PIPE) 

1327 >>> p.name() 

1328 'python' 

1329 >>> p.uids() 

1330 user(real=1000, effective=1000, saved=1000) 

1331 >>> p.username() 

1332 'giampaolo' 

1333 >>> p.communicate() 

1334 ('hi', None) 

1335 >>> p.terminate() 

1336 >>> p.wait(timeout=2) 

1337 0 

1338 >>> 

1339 """

1341 def __init__(self, *args, **kwargs): 

1342 # Explicitly avoid to raise NoSuchProcess in case the process 

1343 # spawned by subprocess.Popen terminates too quickly, see: 

1344 # https://github.com/giampaolo/psutil/issues/193 

1345 self.__subproc = subprocess.Popen(*args, **kwargs) 

1346 self._init(self.__subproc.pid, _ignore_nsp=True) 

1347

1350 

1351 def __enter__(self): 

1352 if hasattr(self.__subproc, '__enter__'): 

1353 self.__subproc.__enter__() 

1354 return self 

1355 

1356 def __exit__(self, *args, **kwargs): 

1357 if hasattr(self.__subproc, '__exit__'): 

1358 return self.__subproc.__exit__(*args, **kwargs) 

1359 else: 

1360 if self.stdout: 

1361 self.stdout.close() 

1362 if self.stderr: 

1363 self.stderr.close() 

1364 try: 

1365 # Flushing a BufferedWriter may raise an error. 

1366 if self.stdin: 

1367 self.stdin.close() 

1368 finally: 

1369 # Wait for the process to terminate, to avoid zombies. 

1370 self.wait()

1424def process_iter(attrs=None, ad_value=None): 

1425 """Return a generator yielding a Process instance for all 

1426 running processes. 

1427 

1428 Every new Process instance is only created once and then cached 

1429 into an internal table which is updated every time this is used. 

1430 

1431 Cached Process instances are checked for identity so that you're 

1432 safe in case a PID has been reused by another process, in which 

1433 case the cached instance is updated. 

1434 

1435 The sorting order in which processes are yielded is based on 

1436 their PIDs. 

1437 

1438 *attrs* and *ad_value* have the same meaning as in 

1439 Process.as_dict(). If *attrs* is specified as_dict() is called 

1440 and the resulting dict is stored as a 'info' attribute attached 

1441 to returned Process instance. 

1442 If *attrs* is an empty list it will retrieve all process info 

1443 (slow). 

1444 """ 

1445 global _pmap 

1446 

1447 def add(pid): 

1448 proc = Process(pid) 

1449 if attrs is not None: 

1450 proc.info = proc.as_dict(attrs=attrs, ad_value=ad_value) 

1451 pmap[proc.pid] = proc 

1452 return proc 

1453 

1454 def remove(pid): 

1455 pmap.pop(pid, None) 

1456 

1457 pmap = _pmap.copy() 

1458 a = set(pids()) 

1459 b = set(pmap.keys()) 

1460 new_pids = a - b 

1461 gone_pids = b - a 

1462 for pid in gone_pids: 

1463 remove(pid) 

1464 try: 

1465 ls = sorted(list(pmap.items()) + list(dict.fromkeys(new_pids).items())) 

1466 for pid, proc in ls: 

1467 try: 

1468 if proc is None: # new process 

1469 yield add(pid) 

1470 else: 

1471 # use is_running() to check whether PID has been 

1472 # reused by another process in which case yield a 

1473 # new Process instance 

1474 if proc.is_running(): 

1475 if attrs is not None: 

1476 proc.info = proc.as_dict( 

1477 attrs=attrs, ad_value=ad_value) 

1478 yield proc 

1479 else: 

1480 yield add(pid) 

1481 except NoSuchProcess: 

1482 remove(pid) 

1483 except AccessDenied: 

1484 # Process creation time can't be determined hence there's 

1485 # no way to tell whether the pid of the cached process 

1486 # has been reused. Just return the cached version. 

1487 if proc is None and pid in pmap: 

1488 try: 

1489 yield pmap[pid] 

1490 except KeyError: 

1491 # If we get here it is likely that 2 threads were 

1492 # using process_iter(). 

1493 pass 

1494 else: 

1495 raise 

1496 finally: 

1497 _pmap = pmap

1500def wait_procs(procs, timeout=None, callback=None): 

1501 """Convenience function which waits for a list of processes to 

1502 terminate. 

1503 

1504 Return a (gone, alive) tuple indicating which processes 

1505 are gone and which ones are still alive. 

1506 

1507 The gone ones will have a new *returncode* attribute indicating 

1508 process exit status (may be None). 

1509 

1510 *callback* is a function which gets called every time a process 

1511 terminates (a Process instance is passed as callback argument). 

1512 

1513 Function will return as soon as all processes terminate or when 

1514 *timeout* occurs. 

1515 Differently from Process.wait() it will not raise TimeoutExpired if 

1516 *timeout* occurs. 

1517 

1518 Typical use case is: 

1519 

1520 - send SIGTERM to a list of processes 

1521 - give them some time to terminate 

1522 - send SIGKILL to those ones which are still alive 

1523 

1524 Example: 

1525 

1526 >>> def on_terminate(proc): 

1527 ... print("process {} terminated".format(proc)) 

1528 ... 

1529 >>> for p in procs: 

1530 ... p.terminate() 

1531 ... 

1532 >>> gone, alive = wait_procs(procs, timeout=3, callback=on_terminate) 

1533 >>> for p in alive: 

1534 ... p.kill() 

1535 """ 

1536 def check_gone(proc, timeout): 

1537 try: 

1538 returncode = proc.wait(timeout=timeout) 

1539 except TimeoutExpired: 

1540 pass 

1541 except _SubprocessTimeoutExpired: 

1542 pass 

1543 else: 

1544 if returncode is not None or not proc.is_running(): 

1545 # Set new Process instance attribute. 

1546 proc.returncode = returncode 

1547 gone.add(proc) 

1548 if callback is not None: 

1549 callback(proc) 

1550 

1551 if timeout is not None and not timeout >= 0: 

1552 msg = "timeout must be a positive integer, got %s" % timeout 

1553 raise ValueError(msg) 

1554 gone = set() 

1555 alive = set(procs) 

1556 if callback is not None and not callable(callback): 

1557 raise TypeError("callback %r is not a callable" % callable) 

1558 if timeout is not None: 

1559 deadline = _timer() + timeout 

1560 

1561 while alive: 

1562 if timeout is not None and timeout <= 0: 

1563 break 

1564 for proc in alive: 

1565 # Make sure that every complete iteration (all processes) 

1566 # will last max 1 sec. 

1567 # We do this because we don't want to wait too long on a 

1568 # single process: in case it terminates too late other 

1569 # processes may disappear in the meantime and their PID 

1570 # reused. 

1571 max_timeout = 1.0 / len(alive) 

1572 if timeout is not None: 

1573 timeout = min((deadline - _timer()), max_timeout) 

1574 if timeout <= 0: 

1575 break 

1576 check_gone(proc, timeout) 

1577 else: 

1578 check_gone(proc, max_timeout) 

1579 alive = alive - gone 

1580 

1581 if alive: 

1582 # Last attempt over processes survived so far. 

1583 # timeout == 0 won't make this function wait any further. 

1584 for proc in alive: 

1585 check_gone(proc, 0) 

1586 alive = alive - gone 

1587 

1588 return (list(gone), list(alive))

1662def _cpu_tot_time(times): 

1663 """Given a cpu_time() ntuple calculates the total CPU time 

1664 (including idle time). 

1665 """ 

1666 tot = sum(times) 

1667 if LINUX: 

1668 # On Linux guest times are already accounted in "user" or 

1669 # "nice" times, so we subtract them from total. 

1670 # Htop does the same. References: 

1671 # https://github.com/giampaolo/psutil/pull/940 

1672 # http://unix.stackexchange.com/questions/178045 

1673 # https://github.com/torvalds/linux/blob/ 

1674 # 447976ef4fd09b1be88b316d1a81553f1aa7cd07/kernel/sched/ 

1675 # cputime.c#L158 

1676 tot -= getattr(times, "guest", 0) # Linux 2.6.24+ 

1677 tot -= getattr(times, "guest_nice", 0) # Linux 3.2.0+ 

1678 return tot

1681def _cpu_busy_time(times): 

1682 """Given a cpu_time() ntuple calculates the busy CPU time. 

1683 We do so by subtracting all idle CPU times. 

1684 """ 

1685 busy = _cpu_tot_time(times) 

1686 busy -= times.idle 

1687 # Linux: "iowait" is time during which the CPU does not do anything 

1688 # (waits for IO to complete). On Linux IO wait is *not* accounted 

1689 # in "idle" time so we subtract it. Htop does the same. 

1690 # References: 

1691 # https://github.com/torvalds/linux/blob/ 

1692 # 447976ef4fd09b1be88b316d1a81553f1aa7cd07/kernel/sched/cputime.c#L244 

1693 busy -= getattr(times, "iowait", 0) 

1694 return busy

1697def _cpu_times_deltas(t1, t2): 

1698 assert t1._fields == t2._fields, (t1, t2) 

1699 field_deltas = [] 

1700 for field in _psplatform.scputimes._fields: 

1701 field_delta = getattr(t2, field) - getattr(t1, field) 

1702 # CPU times are always supposed to increase over time 

1703 # or at least remain the same and that's because time 

1704 # cannot go backwards. 

1705 # Surprisingly sometimes this might not be the case (at 

1706 # least on Windows and Linux), see: 

1707 # https://github.com/giampaolo/psutil/issues/392 

1708 # https://github.com/giampaolo/psutil/issues/645 

1709 # https://github.com/giampaolo/psutil/issues/1210 

1710 # Trim negative deltas to zero to ignore decreasing fields. 

1711 # top does the same. Reference: 

1712 # https://gitlab.com/procps-ng/procps/blob/v3.3.12/top/top.c#L5063 

1713 field_delta = max(0, field_delta) 

1714 field_deltas.append(field_delta) 

1715 return _psplatform.scputimes(*field_deltas)

1718def cpu_percent(interval=None, percpu=False): 

1719 """Return a float representing the current system-wide CPU 

1720 utilization as a percentage. 

1721 

1722 When *interval* is > 0.0 compares system CPU times elapsed before 

1723 and after the interval (blocking). 

1724 

1725 When *interval* is 0.0 or None compares system CPU times elapsed 

1726 since last call or module import, returning immediately (non 

1727 blocking). That means the first time this is called it will 

1728 return a meaningless 0.0 value which you should ignore. 

1729 In this case is recommended for accuracy that this function be 

1730 called with at least 0.1 seconds between calls. 

1731 

1732 When *percpu* is True returns a list of floats representing the 

1733 utilization as a percentage for each CPU. 

1734 First element of the list refers to first CPU, second element 

1735 to second CPU and so on. 

1736 The order of the list is consistent across calls. 

1737 

1738 Examples: 

1739 

1740 >>> # blocking, system-wide 

1741 >>> psutil.cpu_percent(interval=1) 

1742 2.0 

1743 >>> 

1744 >>> # blocking, per-cpu 

1745 >>> psutil.cpu_percent(interval=1, percpu=True) 

1746 [2.0, 1.0] 

1747 >>> 

1748 >>> # non-blocking (percentage since last call) 

1749 >>> psutil.cpu_percent(interval=None) 

1750 2.9 

1751 >>> 

1752 """ 

1753 tid = threading.current_thread().ident 

1754 blocking = interval is not None and interval > 0.0 

1755 if interval is not None and interval < 0: 

1756 raise ValueError("interval is not positive (got %r)" % interval) 

1757 

1758 def calculate(t1, t2): 

1759 times_delta = _cpu_times_deltas(t1, t2) 

1760 all_delta = _cpu_tot_time(times_delta) 

1761 busy_delta = _cpu_busy_time(times_delta) 

1762 

1763 try: 

1764 busy_perc = (busy_delta / all_delta) * 100 

1765 except ZeroDivisionError: 

1766 return 0.0 

1767 else: 

1768 return round(busy_perc, 1) 

1769 

1770 # system-wide usage 

1771 if not percpu: 

1772 if blocking: 

1773 t1 = cpu_times() 

1774 time.sleep(interval) 

1775 else: 

1776 t1 = _last_cpu_times.get(tid) or cpu_times() 

1777 _last_cpu_times[tid] = cpu_times() 

1778 return calculate(t1, _last_cpu_times[tid]) 

1779 # per-cpu usage 

1780 else: 

1781 ret = [] 

1782 if blocking: 

1783 tot1 = cpu_times(percpu=True) 

1784 time.sleep(interval) 

1785 else: 

1786 tot1 = _last_per_cpu_times.get(tid) or cpu_times(percpu=True) 

1787 _last_per_cpu_times[tid] = cpu_times(percpu=True) 

1788 for t1, t2 in zip(tot1, _last_per_cpu_times[tid]): 

1789 ret.append(calculate(t1, t2)) 

1790 return ret 

1791

1793# Use a separate dict for cpu_times_percent(), so it's independent from 

1794# cpu_percent() and they can both be used within the same program. 

1795_last_cpu_times_2 = _last_cpu_times.copy() 

1796_last_per_cpu_times_2 = _last_per_cpu_times.copy()

1800 """Same as cpu_percent() but provides utilization percentages 

1801 for each specific CPU time as is returned by cpu_times(). 

1802 For instance, on Linux we'll get: 

1803 

1804 >>> cpu_times_percent() 

1805 cpupercent(user=4.8, nice=0.0, system=4.8, idle=90.5, iowait=0.0, 

1806 irq=0.0, softirq=0.0, steal=0.0, guest=0.0, guest_nice=0.0) 

1807 >>> 

1808 

1809 *interval* and *percpu* arguments have the same meaning as in 

1810 cpu_percent(). 

1811 """

1813 blocking = interval is not None and interval > 0.0 

1814 if interval is not None and interval < 0: 

1815 raise ValueError("interval is not positive (got %r)" % interval)

1826 for field_delta in times_delta:

1828 field_perc = round(field_perc, 1) 

1829 # make sure we don't return negative values or values over 100% 

1830 field_perc = min(max(0.0, field_perc), 100.0) 

1831 nums.append(field_perc) 

1832 return _psplatform.scputimes(*nums) 

1833 

1834 # system-wide usage 

1835 if not percpu: 

1836 if blocking: 

1837 t1 = cpu_times() 

1838 time.sleep(interval) 

1839 else: 

1840 t1 = _last_cpu_times_2.get(tid) or cpu_times() 

1841 _last_cpu_times_2[tid] = cpu_times() 

1842 return calculate(t1, _last_cpu_times_2[tid])

1844 else: 

1845 ret = [] 

1846 if blocking: 

1847 tot1 = cpu_times(percpu=True) 

1848 time.sleep(interval) 

1849 else: 

1850 tot1 = _last_per_cpu_times_2.get(tid) or cpu_times(percpu=True) 

1851 _last_per_cpu_times_2[tid] = cpu_times(percpu=True) 

1852 for t1, t2 in zip(tot1, _last_per_cpu_times_2[tid]): 

1853 ret.append(calculate(t1, t2)) 

1854 return ret

1923def virtual_memory(): 

1924 """Return statistics about system memory usage as a namedtuple 

1925 including the following fields, expressed in bytes: 

1926 

1927 - total: 

1928 total physical memory available. 

1929 

1930 - available: 

1931 the memory that can be given instantly to processes without the 

1932 system going into swap. 

1933 This is calculated by summing different memory values depending 

1934 on the platform and it is supposed to be used to monitor actual 

1935 memory usage in a cross platform fashion. 

1936 

1937 - percent: 

1938 the percentage usage calculated as (total - available) / total * 100 

1939 

1940 - used: 

1941 memory used, calculated differently depending on the platform and 

1942 designed for informational purposes only: 

1943 macOS: active + wired 

1944 BSD: active + wired + cached 

1945 Linux: total - free 

1946 

1947 - free:

2169def net_if_addrs():

2178 - netmask: and 'broadcast' and 'ptp' may be None.

2180 destination address on a point to point interface
