80def If(*args): 

81 # the coercion here is strange enough that we'll just implement it manually 

82 if len(args) != 3: 

83 raise ClaripyOperationError("invalid number of args passed to If") 

84 

85 args = list(args) 

86 

87 if isinstance(args[0], bool): 

88 args[0] = BoolV(args[0]) 

89 

90 ty = None 

91 if isinstance(args[1], Base): 

92 ty = type(args[1]) 

93 elif isinstance(args[2], Base): 

94 ty = type(args[2]) 

95 else: 

96 raise ClaripyTypeError("true/false clause of If must have bearable types") 

97 

98 if isinstance(args[1], Bits) and isinstance(args[2], Bits) and args[1].length != args[2].length: 

99 raise ClaripyTypeError("sized arguments to If must have the same length") 

100 

101 if not isinstance(args[1], ty): 

102 if hasattr(ty, '_from_' + type(args[1]).__name__): 

103 convert = getattr(ty, '_from_' + type(args[1]).__name__) 

104 args[1] = convert(args[2], args[1]) 

105 else: 

106 raise ClaripyTypeError("can't convert {} to {}".format(type(args[1]), ty)) 

107 if not isinstance(args[2], ty): 

108 if hasattr(ty, '_from_' + type(args[2]).__name__): 

109 convert = getattr(ty, '_from_' + type(args[2]).__name__) 

110 args[2] = convert(args[1], args[2]) 

111 else: 

112 raise ClaripyTypeError("can't convert {} to {}".format(type(args[2]), ty)) 

113 

114 if is_true(args[0]): 

115 return args[1].append_annotations(args[0].annotations) 

116 elif is_false(args[0]): 

117 return args[2].append_annotations(args[0].annotations) 

118 

119 if isinstance(args[1], Base) and args[1].op == 'If' and args[1].args[0] is args[0]: 

120 return If(args[0], args[1].args[1], args[2]) 

121 if isinstance(args[1], Base) and args[1].op == 'If' and args[1].args[0] is Not(args[0]): 

122 return If(args[0], args[1].args[2], args[2]) 

123 if isinstance(args[2], Base) and args[2].op == 'If' and args[2].args[0] is args[0]: 

124 return If(args[0], args[1], args[2].args[2]) 

125 if isinstance(args[2], Base) and args[2].op == 'If' and args[2].args[0] is Not(args[0]): 

126 return If(args[0], args[1], args[2].args[1]) 

127 

128 if args[1] is args[2]: 

129 return args[1] 

130 if args[1] is true and args[2] is false: 

131 return args[0] 

132 if args[1] is false and args[2] is true: 

133 return ~args[0] 

134 

135 if issubclass(ty, Bits): 

136 return ty('If', tuple(args), length=args[1].length) 

137 else: 

138 return ty('If', tuple(args))

166#For large tables, ite_dict that uses a binary search tree instead of a "linear" search tree. 

167#This improves Z3 search capability (eliminating branches) and decreases recursion depth: 

168#linear search trees make Z3 error out on tables larger than a couple hundred elements.) 

169def ite_dict(i, d, default): 

170 

171 """ 

172 Return an expression of if-then-else trees which expresses a switch tree 

173 :param i: The variable which may take on multiple values affecting the final result 

174 :param d: A dict mapping possible values for i to values which the result could be 

175 :param default: A default value that the expression should take on if `i` matches none of the keys of `d` 

176 :return: An expression encoding the result of the above 

177 """ 

178 i = i.ast if type(i) is ASTCacheKey else i 

179 

180 #for small dicts fall back to the linear implementation 

181 if len(d) < 4: 

182 return ite_cases([ (i == c, v) for c,v in d.items() ], default) 

183 

184 #otherwise, binary search. 

185 #Find the median: 

186 keys = list(d.keys()) 

187 keys.sort() 

188 split_val = keys[len(keys)//2] 

189 

190 #split the dictionary 

191 dictLow = {c:v for c,v in d.items() if c <= split_val} 

192 dictHigh = {c:v for c,v in d.items() if c > split_val} 

193 

194 valLow = ite_dict(i, dictLow, default) 

195 valHigh = ite_dict(i, dictHigh, default) 

196 return If(i <= split_val, valLow, valHigh)

198def ite_cases(cases, default): 

199 """ 

200 Return an expression of if-then-else trees which expresses a series of alternatives 

201 

202 :param cases: A list of tuples (c, v). `c` is the condition under which `v` should be the result of the expression 

203 :param default: A default value that the expression should take on if none of the `c` conditions are satisfied 

204 :return: An expression encoding the result of the above 

205 """ 

206 sofar = default 

207 for c,v in reversed(list(cases)): 

208 if is_true(v == sofar): 

209 continue 

210 sofar = If(c, v, sofar) 

211 return sofar

213def reverse_ite_cases(ast): 

214 """ 

215 Given an expression created by `ite_cases`, produce the cases that generated it 

216 :param ast: 

217 :return: 

218 """ 

219 queue = [(true, ast)] 

220 while queue: 

221 condition, ast = queue.pop(0) 

222 if ast.op == 'If': 

223 queue.append((And(condition, ast.args[0]), ast.args[1])) 

224 queue.append((And(condition, Not(ast.args[0])), ast.args[2])) 

225 else: 

226 yield condition, ast

231 

232 :param expr: 

233 :return: 

234 """ 

235
