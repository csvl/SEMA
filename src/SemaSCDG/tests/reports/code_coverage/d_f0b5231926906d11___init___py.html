27class SimLibrary: 

28 """ 

29 A SimLibrary is the mechanism for describing a dynamic library's API, its functions and metadata. 

30 

31 Any instance of this class (or its subclasses) found in the ``angr.procedures.definitions`` package will be 

32 automatically picked up and added to ``angr.SIM_LIBRARIES`` via all its names. 

33 

34 :ivar fallback_cc: A mapping from architecture to the default calling convention that should be used if no 

35 other information is present. Contains some sane defaults for linux. 

36 :ivar fallback_proc: A SimProcedure class that should be used to provide stub procedures. By default, 

37 ``ReturnUnconstrained``.

184 def _apply_metadata(self, proc, arch): 

185 if proc.cc is None and arch.name in self.default_ccs: 

186 proc.cc = self.default_ccs[arch.name](arch) 

187 if proc.cc is None and arch.name in self.fallback_cc:

189 if proc.display_name in self.prototypes: 

190 proc.prototype = self.prototypes[proc.display_name].with_arch(arch) 

191 if proc.prototype.arg_names is None: 

192 # Use inspect to extract the parameters from the run python function 

193 proc.prototype.arg_names = inspect.getfullargspec(proc.run).args[1:] 

194 if not proc.ARGS_MISMATCH: 

195 proc.num_args = len(proc.prototype.args) 

196 if proc.display_name in self.non_returning: 

197 proc.returns = False 

198 proc.library_name = self.name

280class SimCppLibrary(SimLibrary): 

281 """ 

282 SimCppLibrary is a specialized version of SimLibrary that will demangle C++ function names before looking for an 

283 implementation or prototype for it. 

284 """

286 @staticmethod 

287 def _try_demangle(name): 

288 if name[0:2] == "_Z": 

289 try: 

290 ast = itanium_demangler.parse(name) 

291 except NotImplementedError: 

292 return name 

293 if ast: 

294 return str(ast) 

295 return name

297 @staticmethod 

298 def _proto_from_demangled_name(name: str) -> Optional[SimTypeFunction]: 

299 """ 

300 Attempt to extract arguments and calling convention information for a C++ function whose name was mangled 

301 according to the Itanium C++ ABI symbol mangling language. 

302 

303 :param name: The demangled function name. 

304 :return: A prototype or None if a prototype cannot be found. 

305 """ 

306 

307 try: 

308 parsed, _ = parse_cpp_file(name, with_param_names=False) 

309 except ValueError: 

310 return None 

311 if not parsed: 

312 return None 

313 _, func_proto = next(iter(parsed.items())) 

314 return func_proto

316 def get(self, name, arch): 

317 """ 

318 Get an implementation of the given function specialized for the given arch, or a stub procedure if none exists. 

319 Demangle the function name if it is a mangled C++ name. 

320 

321 :param str name: The name of the function as a string 

322 :param arch: The architecure to use, as either a string or an archinfo.Arch instance 

323 :return: A SimProcedure instance representing the function as found in the library 

324 """ 

325 demangled_name = self._try_demangle(name) 

326 if demangled_name not in self.procedures: 

327 return self.get_stub(name, arch) # get_stub() might use the mangled name to derive the function prototype 

328 return super().get(demangled_name, arch)

330 def get_stub(self, name, arch): 

331 """ 

332 Get a stub procedure for the given function, regardless of if a real implementation is available. This will 

333 apply any metadata, such as a default calling convention or a function prototype. Demangle the function name 

334 if it is a mangled C++ name. 

335 

336 :param str name: The name of the function as a string 

337 :param arch: The architecture to use, as either a string or an archinfo.Arch instance 

338 :return: A SimProcedure instance representing a plausable stub as could be found in the library. 

339 """ 

340 demangled_name = self._try_demangle(name) 

341 stub = super().get_stub(demangled_name, arch) 

342 # try to determine a prototype from the function name if possible 

343 if demangled_name != name: 

344 # itanium-mangled function name 

345 stub.prototype = self._proto_from_demangled_name(demangled_name) 

346 if stub.prototype is not None: 

347 stub.prototype = stub.prototype.with_arch(arch) 

348 if not stub.ARGS_MISMATCH: 

349 stub.cc.num_args = len(stub.prototype.args) 

350 stub.num_args = len(stub.prototype.args) 

351 return stub

396class SimSyscallLibrary(SimLibrary): 

397 """ 

398 SimSyscallLibrary is a specialized version of SimLibrary for dealing not with a dynamic library's API but rather 

399 an operating system's syscall API. Because this interface is inherently lower-level than a dynamic library, many 

400 parts of this class has been changed to store data based on an "ABI name" (ABI = application binary interface, 

401 like an API but for when there's no programming language) instead of an architecture. An ABI name is just an 

402 arbitrary string with which a calling convention and a syscall numbering is associated. 

403 

404 All the SimLibrary methods for adding functions still work, but now there's an additional layer on top that 

405 associates them with numbers. 

406 """ 

407 def __init__(self): 

408 super().__init__() 

409 self.syscall_number_mapping: Dict[str,Dict[int,str]] = defaultdict(dict) # keyed by abi 

410 self.syscall_name_mapping: Dict[str,Dict[str,int]] = defaultdict(dict) # keyed by abi 

411 self.default_cc_mapping: Dict[str,Type['SimCCSyscall']] = {} # keyed by abi 

412 self.syscall_prototypes: Dict[str,Dict[str,SimTypeFunction]] = defaultdict(dict) # keyed by abi 

413 self.fallback_proc = stub_syscall 

414 

415 def copy(self): 

416 o = SimSyscallLibrary() 

417 o.procedures = dict(self.procedures) 

418 o.non_returning = set(self.non_returning) 

419 o.prototypes = dict(self.prototypes) 

420 o.default_ccs = dict(self.default_ccs) 

421 o.names = list(self.names) 

422 o.syscall_number_mapping = defaultdict(dict, self.syscall_number_mapping) # {abi: {number: name}} 

423 o.syscall_name_mapping = defaultdict(dict, self.syscall_name_mapping) # {abi: {name: number}} 

424 o.syscall_prototypes = defaultdict(dict, self.syscall_prototypes) # as above 

425 o.default_cc_mapping = dict(self.default_cc_mapping) # {abi: cc} 

426 return o

503 def _canonicalize(self, number, arch, abi_list): 

504 if type(arch) is str: 

505 arch = archinfo.arch_from_id(arch) 

506 if type(number) is str: 

507 return number, arch, None 

508 for abi in abi_list: 

509 mapping = self.syscall_number_mapping[abi] 

510 if number in mapping: 

511 return mapping[number], arch, abi 

512 return 'sys_%d' % number, arch, None

523 proc.prototype = self.syscall_prototypes[abi][name].with_arch(arch)

526 def get(self, number, arch, abi_list=()): 

527 """ 

528 The get() function for SimSyscallLibrary looks a little different from its original version. 

529 

530 Instead of providing a name, you provide a number, and you additionally provide a list of abi names that are 

531 applicable. The first abi for which the number is present in the mapping will be chosen. This allows for the 

532 easy abstractions of architectures like ARM or MIPS linux for which there are many ABIs that can be used at any 

533 time by using syscall numbers from various ranges. If no abi knows about the number, the stub procedure with 

534 the name "sys_%d" will be used. 

535 

536 :param number: The syscall number 

537 :param arch: The architecture being worked with, as either a string name or an archinfo.Arch 

538 :param abi_list: A list of ABI names that could be used 

539 :return: A SimProcedure representing the implementation of the given syscall, or a stub if no 

540 implementation is available 

541 """ 

542 name, arch, abi = self._canonicalize(number, arch, abi_list) 

543 proc = super().get(name, arch) 

544 proc.is_syscall = True 

545 self._apply_numerical_metadata(proc, number, arch, abi) 

546 return proc

604 def has_prototype(self, abi: str, name: str) -> bool: 

605 """ 

606 Check if a function has a prototype associated with it. Demangle the function name if it is a mangled C++ name. 

607 

608 :param abi: Name of the ABI. 

609 :param name: The syscall name. 

610 :return: bool 

611 """ 

612 if abi not in self.syscall_prototypes: 

613 return False 

614 return name in self.syscall_prototypes[abi]
