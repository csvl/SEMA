97 def state_blank(self, addr=None, initial_prefix=None, brk=None, stack_end=None, stack_size=1024*1024*8, stdin=None, 

98 thread_idx=None, permissions_backer=None, **kwargs): 

99 """ 

100 Initialize a blank state. 

101 

102 All parameters are optional. 

103 

104 :param addr: The execution start address. 

105 :param initial_prefix: 

106 :param stack_end: The end of the stack (i.e., the byte after the last valid stack address). 

107 :param stack_size: The number of bytes to allocate for stack space 

108 :param brk: The address of the process' break. 

109 :return: The initialized SimState. 

110 

111 Any additional arguments will be passed to the SimState constructor 

112 """ 

113 # TODO: move ALL of this into the SimState constructor 

114 if kwargs.get('mode', None) is None: 

115 kwargs['mode'] = self.project._default_analysis_mode 

116 if permissions_backer is not None: 

117 kwargs['permissions_map'] = permissions_backer[1] 

118 kwargs['default_permissions'] = 7 if permissions_backer[0] else 3 

119 if kwargs.get('cle_memory_backer', None) is None: 

120 kwargs['cle_memory_backer'] = self.project.loader 

121 if kwargs.get('os_name', None) is None: 

122 kwargs['os_name'] = self.name 

123 actual_stack_end = stack_end 

124 if stack_end is None: 

125 stack_end = self.arch.initial_sp 

126 

127 if kwargs.get('permissions_map', None) is None: 

128 # just a dict of address ranges to permission bits 

129 permission_map = { } 

130 for obj in self.project.loader.all_objects: 

131 for seg in obj.segments: 

132 perms = 0 

133 # bit values based off of protection bit values from sys/mman.h 

134 if seg.is_readable: 

135 perms |= 1 # PROT_READ 

136 if seg.is_writable: 

137 perms |= 2 # PROT_WRITE 

138 if seg.is_executable: 

139 perms |= 4 # PROT_EXEC 

140 permission_map[(seg.min_addr, seg.max_addr)] = perms 

141 kwargs['permissions_map'] = permission_map 

142 if self.project.loader.main_object.execstack: 

143 stack_perms = 1 | 2 | 4 # RWX 

144 else: 

145 stack_perms = 1 | 2 # RW 

146 

147 state = SimState(self.project, stack_end=stack_end, stack_size=stack_size, stack_perms=stack_perms, **kwargs) 

148 

149 if stdin is not None and not isinstance(stdin, SimFileBase): 

150 if type(stdin) is type: 

151 stdin = stdin(name='stdin', has_end=False) 

152 else: 

153 if isinstance(stdin, claripy.Bits): 

154 num_bytes = len(stdin) // self.project.arch.byte_width 

155 else: 

156 num_bytes = len(stdin) 

157 _l.warning("stdin is constrained to %d bytes (has_end=True). If you are only providing the first " 

158 "%d bytes instead of the entire stdin, please use " 

159 "stdin=SimFileStream(name='stdin', content=your_first_n_bytes, has_end=False).", 

160 num_bytes, num_bytes) 

161 stdin = SimFileStream(name='stdin', content=stdin, has_end=True) 

162 

163 last_addr = self.project.loader.main_object.max_addr 

164 actual_brk = (last_addr - last_addr % 0x1000 + 0x1000) if brk is None else brk 

165 state.register_plugin('posix', SimSystemPosix(stdin=stdin, brk=actual_brk)) 

166 

167 if initial_prefix is not None: 

168 for reg in state.arch.default_symbolic_registers: 

169 state.registers.store(reg, state.solver.BVS( 

170 initial_prefix + "_" + reg, 

171 state.arch.bits, 

172 explicit_name=True, 

173 key=('reg', reg), 

174 eternal=True)) 

175 

176 if state.arch.sp_offset is not None: 

177 state.regs.sp = stack_end 

178 

179 for reg, val, is_addr, mem_region in state.arch.default_register_values: 

180 region_base = None # so pycharm does not complain 

181 

182 if is_addr: 

183 if isinstance(mem_region, tuple): 

184 # unpack it 

185 mem_region, region_base = mem_region 

186 elif mem_region == 'global': 

187 # Backward compatibility 

188 region_base = 0 

189 else: 

190 raise AngrSimOSError('You must specify the base address for memory region "%s". ' % mem_region) 

191 

192 # special case for stack_end overriding sp default 

193 if actual_stack_end is not None and state.arch.registers[reg][0] == state.arch.sp_offset: 

194 continue 

195 

196 if o.ABSTRACT_MEMORY in state.options and is_addr: 

197 address = claripy.ValueSet(state.arch.bits, mem_region, region_base, val) 

198 state.registers.store(reg, address) 

199 else: 

200 state.registers.store(reg, val) 

201 

202 if addr is None: 

203 state.regs.ip = self.project.entry 

204 

205 thread_name = self.project.loader.main_object.threads[thread_idx] if thread_idx is not None else None 

206 for reg, val in self.project.loader.main_object.thread_registers(thread_name).items(): 

207 if reg in ('fs', 'gs', 'cs', 'ds', 'es', 'ss') and state.arch.name == 'X86': 

208 state.registers.store(reg, val >> 16) # oh boy big hack 

209 elif reg in state.arch.registers or reg in ('flags', 'eflags', 'rflags'): 

210 state.registers.store(reg, val) 

211 elif reg == 'fctrl': 

212 state.regs.fpround = (val & 0xC00) >> 10 

213 elif reg == 'fstat': 

214 state.regs.fc3210 = (val & 0x4700) 

215 elif reg == 'ftag': 

216 empty_bools = [((val >> (x * 2)) & 3) == 3 for x in range(8)] 

217 tag_chars = [claripy.BVV(0 if x else 1, 8) for x in empty_bools] 

218 for i, tag in enumerate(tag_chars): 

219 setattr(state.regs, 'fpu_t%d' % i, tag) 

220 elif reg in ('fiseg', 'fioff', 'foseg', 'fooff', 'fop'): 

221 pass 

222 elif reg == 'mxcsr': 

223 state.regs.sseround = (val & 0x600) >> 9 

224 else: 

225 _l.error("What is this register %s I have to translate?", reg) 

226 

227 

228 if addr is not None: 

229 state.regs.ip = addr 

230 

231 # set up the "root history" node 

232 state.scratch.ins_addr = addr 

233 state.scratch.bbl_addr = addr 

234 state.scratch.stmt_idx = 0 

235 state.history.jumpkind = 'Ijk_Boring' 

236 

237 return state

270 def prepare_call_state(self, calling_state, initial_state=None, 

271 preserve_registers=(), preserve_memory=()): 

272 """ 

273 This function prepares a state that is executing a call instruction. 

274 If given an initial_state, it copies over all of the critical registers to it from the 

275 calling_state. Otherwise, it prepares the calling_state for action. 

276 

277 This is mostly used to create minimalistic for CFG generation. Some ABIs, such as MIPS PIE and 

278 x86 PIE, require certain information to be maintained in certain registers. For example, for 

279 PIE MIPS, this function transfer t9, gp, and ra to the new state. 

280 """ 

281 

282 if isinstance(self.arch, ArchMIPS32): 

283 if initial_state is not None: 

284 initial_state = self.state_blank() 

285 mips_caller_saves = ('s0', 's1', 's2', 's3', 's4', 's5', 's6', 's7', 'gp', 'sp', 'bp', 'ra') 

286 preserve_registers = preserve_registers + mips_caller_saves + ('t9',) 

287 

288 if initial_state is None: 

289 new_state = calling_state.copy() 

290 else: 

291 new_state = initial_state.copy() 

292 for reg in set(preserve_registers): 

293 new_state.registers.store(reg, calling_state.registers.load(reg)) 

294 for addr, val in set(preserve_memory): 

295 new_state.memory.store(addr, calling_state.memory.load(addr, val)) 

296 

297 return new_state

361 def generate_gdt(self, fs, gs, fs_size=0xFFFFFFFF, gs_size=0xFFFFFFFF): 

362 """ 

363 Generate a GlobalDescriptorTable object and populate it using the value of the gs and fs register 

364 

365 :param fs: value of the fs segment register 

366 :param gs: value of the gs segment register 

367 :param fs_size: size of the fs segment register 

368 :param gs_size: size of the gs segment register 

369 :return: gdt a GlobalDescriptorTable object 

370 """ 

371 A_PRESENT = 0x80 

372 A_DATA = 0x10 

373 A_DATA_WRITABLE = 0x2 

374 A_PRIV_0 = 0x0 

375 A_DIR_CON_BIT = 0x4 

376 F_PROT_32 = 0x4 

377 S_GDT = 0x0 

378 S_PRIV_0 = 0x0 

379 GDT_ADDR = 0x4000 

380 GDT_LIMIT = 0x1000 

381 

382 normal_entry = self._create_gdt_entry(0, 0xFFFFFFFF, 

383 A_PRESENT | A_DATA | A_DATA_WRITABLE | A_PRIV_0 | A_DIR_CON_BIT, 

384 F_PROT_32) 

385 stack_entry = self._create_gdt_entry(0, 0xFFFFFFFF, A_PRESENT | A_DATA | A_DATA_WRITABLE | A_PRIV_0, 

386 F_PROT_32) 

387 fs_entry = self._create_gdt_entry(fs, fs_size, 

388 A_PRESENT | A_DATA | A_DATA_WRITABLE | A_PRIV_0 | A_DIR_CON_BIT, F_PROT_32) 

389 gs_entry = self._create_gdt_entry(gs, gs_size, 

390 A_PRESENT | A_DATA | A_DATA_WRITABLE | A_PRIV_0 | A_DIR_CON_BIT, F_PROT_32) 

391 

392 table = normal_entry + stack_entry + fs_entry + gs_entry 

393 gdt = (GDT_ADDR << 16 | GDT_LIMIT) 

394 selector = self._create_selector(1, S_GDT | S_PRIV_0) 

395 cs = selector 

396 ds = selector 

397 es = selector 

398 selector = self._create_selector(2, S_GDT | S_PRIV_0) 

399 ss = selector 

400 selector = self._create_selector(3, S_GDT | S_PRIV_0) 

401 fs = selector 

402 selector = self._create_selector(4, S_GDT | S_PRIV_0) 

403 gs = selector 

404 global_descriptor_table = GlobalDescriptorTable(GDT_ADDR, GDT_LIMIT, table, gdt, cs, ds, es, ss, fs, gs) 

405 return global_descriptor_table

407 @staticmethod 

408 def _create_selector(idx, flags): 

409 to_ret = flags 

410 to_ret |= idx << 3 

411 return to_ret 

412 

413 @staticmethod 

414 def _create_gdt_entry(base, limit, access, flags): 

415 to_ret = limit & 0xffff 

416 to_ret |= (base & 0xffffff) << 16 

417 to_ret |= (access & 0xff) << 40 

418 to_ret |= ((limit >> 16) & 0xf) << 48 

419 to_ret |= (flags & 0xff) << 52 

420 to_ret |= ((base >> 24) & 0xff) << 56 

421 return struct.pack('<Q', to_ret)
