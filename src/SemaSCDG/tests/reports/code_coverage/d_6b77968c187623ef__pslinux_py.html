207def readlink(path): 

208 """Wrapper around os.readlink().""" 

209 assert isinstance(path, basestring), path 

210 path = os.readlink(path) 

211 # readlink() might return paths containing null bytes ('\x00') 

212 # resulting in "TypeError: must be encoded string without NULL 

213 # bytes, not str" errors when the string is passed to other 

214 # fs-related functions (os.*, open(), ...). 

215 # Apparently everything after '\x00' is garbage (we can have 

216 # ' (deleted)', 'new' and possibly others), see: 

217 # https://github.com/giampaolo/psutil/issues/717 

218 path = path.split('\x00')[0] 

219 # Certain paths have ' (deleted)' appended. Usually this is 

220 # bogus as the file actually exists. Even if it doesn't we 

221 # don't care. 

222 if path.endswith(' (deleted)') and not path_exists_strict(path): 

223 path = path[:-10] 

224 return path

227def file_flags_to_mode(flags): 

228 """Convert file's open() flags into a readable string. 

229 Used by Process.open_files(). 

230 """ 

231 modes_map = {os.O_RDONLY: 'r', os.O_WRONLY: 'w', os.O_RDWR: 'w+'} 

232 mode = modes_map[flags & (os.O_RDONLY | os.O_WRONLY | os.O_RDWR)] 

233 if flags & os.O_APPEND: 

234 mode = mode.replace('w', 'a', 1) 

235 mode = mode.replace('w+', 'r+') 

236 # possible values: r, w, a, r+, a+ 

237 return mode

240def is_storage_device(name): 

241 """Return True if the given name refers to a root device (e.g. 

242 "sda", "nvme0n1") as opposed to a logical partition (e.g. "sda1", 

243 "nvme0n1p1"). If name is a virtual device (e.g. "loop1", "ram") 

244 return True. 

245 """ 

246 # Re-adapted from iostat source code, see: 

247 # https://github.com/sysstat/sysstat/blob/ 

248 # 97912938cd476645b267280069e83b1c8dc0e1c7/common.c#L208 

249 # Some devices may have a slash in their name (e.g. cciss/c0d0...). 

250 name = name.replace('/', '!') 

251 including_virtual = True 

252 if including_virtual: 

253 path = "/sys/block/%s" % name 

254 else: 

255 path = "/sys/block/%s/device" % name 

256 return os.access(path, os.F_OK)

345def calculate_avail_vmem(mems): 

346 """Fallback for kernels < 3.14 where /proc/meminfo does not provide 

347 "MemAvailable", see: 

348 https://blog.famzah.net/2014/09/24/. 

349 

350 This code reimplements the algorithm outlined here: 

351 https://git.kernel.org/cgit/linux/kernel/git/torvalds/linux.git/ 

352 commit/?id=34e431b0ae398fc54ea69ff85ec700722c9da773 

353 

354 We use this function also when "MemAvailable" returns 0 (possibly a 

355 kernel bug, see: https://github.com/giampaolo/psutil/issues/1915). 

356 In that case this routine matches "free" CLI tool result ("available" 

357 column). 

358 

359 XXX: on recent kernels this calculation may differ by ~1.5% compared 

360 to "MemAvailable:", as it's calculated slightly differently. 

361 It is still way more realistic than doing (free + cached) though. 

362 See: 

363 * https://gitlab.com/procps-ng/procps/issues/42 

364 * https://github.com/famzah/linux-memavailable-procfs/issues/2 

365 """ 

366 # Note about "fallback" value. According to: 

367 # https://git.kernel.org/cgit/linux/kernel/git/torvalds/linux.git/ 

368 # commit/?id=34e431b0ae398fc54ea69ff85ec700722c9da773 

369 # ...long ago "available" memory was calculated as (free + cached), 

370 # We use fallback when one of these is missing from /proc/meminfo: 

371 # "Active(file)": introduced in 2.6.28 / Dec 2008 

372 # "Inactive(file)": introduced in 2.6.28 / Dec 2008 

373 # "SReclaimable": introduced in 2.6.19 / Nov 2006 

374 # /proc/zoneinfo: introduced in 2.6.13 / Aug 2005 

375 free = mems[b'MemFree:'] 

376 fallback = free + mems.get(b"Cached:", 0) 

377 try: 

378 lru_active_file = mems[b'Active(file):'] 

379 lru_inactive_file = mems[b'Inactive(file):'] 

380 slab_reclaimable = mems[b'SReclaimable:'] 

381 except KeyError as err: 

382 debug("%r is missing from /proc/meminfo; using an approximation " 

383 "for calculating available memory" % err.args[0]) 

384 return fallback 

385 try: 

386 f = open_binary('%s/zoneinfo' % get_procfs_path()) 

387 except IOError: 

388 return fallback # kernel 2.6.13 

389 

390 watermark_low = 0 

391 with f: 

392 for line in f: 

393 line = line.strip() 

394 if line.startswith(b'low'): 

395 watermark_low += int(line.split()[1]) 

396 watermark_low *= PAGESIZE 

397 

398 avail = free - watermark_low 

399 pagecache = lru_active_file + lru_inactive_file 

400 pagecache -= min(pagecache / 2, watermark_low) 

401 avail += pagecache 

402 avail += slab_reclaimable - min(slab_reclaimable / 2.0, watermark_low) 

403 return int(avail)

406def virtual_memory(): 

407 """Report virtual memory stats. 

408 This implementation mimicks procps-ng-3.3.12, aka "free" CLI tool: 

409 https://gitlab.com/procps-ng/procps/blob/ 

410 24fd2605c51fccc375ab0287cec33aa767f06718/proc/sysinfo.c#L778-791 

411 The returned values are supposed to match both "free" and "vmstat -s" 

412 CLI tools. 

413 """ 

414 missing_fields = [] 

415 mems = {} 

416 with open_binary('%s/meminfo' % get_procfs_path()) as f: 

417 for line in f: 

418 fields = line.split() 

419 mems[fields[0]] = int(fields[1]) * 1024 

420 

421 # /proc doc states that the available fields in /proc/meminfo vary 

422 # by architecture and compile options, but these 3 values are also 

423 # returned by sysinfo(2); as such we assume they are always there. 

424 total = mems[b'MemTotal:'] 

425 free = mems[b'MemFree:'] 

426 try: 

427 buffers = mems[b'Buffers:'] 

428 except KeyError: 

429 # https://github.com/giampaolo/psutil/issues/1010 

430 buffers = 0 

431 missing_fields.append('buffers') 

432 try: 

433 cached = mems[b"Cached:"] 

434 except KeyError: 

435 cached = 0 

436 missing_fields.append('cached') 

437 else: 

438 # "free" cmdline utility sums reclaimable to cached. 

439 # Older versions of procps used to add slab memory instead. 

440 # This got changed in: 

441 # https://gitlab.com/procps-ng/procps/commit/ 

442 # 05d751c4f076a2f0118b914c5e51cfbb4762ad8e 

443 cached += mems.get(b"SReclaimable:", 0) # since kernel 2.6.19 

444 

445 try: 

446 shared = mems[b'Shmem:'] # since kernel 2.6.32 

447 except KeyError: 

448 try: 

449 shared = mems[b'MemShared:'] # kernels 2.4 

450 except KeyError: 

451 shared = 0 

452 missing_fields.append('shared') 

453 

454 try: 

455 active = mems[b"Active:"] 

456 except KeyError: 

457 active = 0 

458 missing_fields.append('active') 

459 

460 try: 

461 inactive = mems[b"Inactive:"] 

462 except KeyError: 

463 try: 

464 inactive = \ 

465 mems[b"Inact_dirty:"] + \ 

466 mems[b"Inact_clean:"] + \ 

467 mems[b"Inact_laundry:"] 

468 except KeyError: 

469 inactive = 0 

470 missing_fields.append('inactive') 

471 

472 try: 

473 slab = mems[b"Slab:"] 

474 except KeyError: 

475 slab = 0 

476 

477 used = total - free - cached - buffers 

478 if used < 0: 

479 # May be symptomatic of running within a LCX container where such 

480 # values will be dramatically distorted over those of the host. 

481 used = total - free 

482 

483 # - starting from 4.4.0 we match free's "available" column. 

484 # Before 4.4.0 we calculated it as (free + buffers + cached) 

485 # which matched htop. 

486 # - free and htop available memory differs as per: 

487 # http://askubuntu.com/a/369589 

488 # http://unix.stackexchange.com/a/65852/168884 

489 # - MemAvailable has been introduced in kernel 3.14 

490 try: 

491 avail = mems[b'MemAvailable:'] 

492 except KeyError: 

493 avail = calculate_avail_vmem(mems) 

494 else: 

495 if avail == 0: 

496 # Yes, it can happen (probably a kernel bug): 

497 # https://github.com/giampaolo/psutil/issues/1915 

498 # In this case "free" CLI tool makes an estimate. We do the same, 

499 # and it matches "free" CLI tool. 

500 avail = calculate_avail_vmem(mems) 

501 

502 if avail < 0: 

503 avail = 0 

504 missing_fields.append('available') 

505 elif avail > total: 

506 # If avail is greater than total or our calculation overflows, 

507 # that's symptomatic of running within a LCX container where such 

508 # values will be dramatically distorted over those of the host. 

509 # https://gitlab.com/procps-ng/procps/blob/ 

510 # 24fd2605c51fccc375ab0287cec33aa767f06718/proc/sysinfo.c#L764 

511 avail = free 

512 

513 percent = usage_percent((total - avail), total, round_=1) 

514 

515 # Warn about missing metrics which are set to 0. 

516 if missing_fields: 

517 msg = "%s memory stats couldn't be determined and %s set to 0" % ( 

518 ", ".join(missing_fields), 

519 "was" if len(missing_fields) == 1 else "were") 

520 warnings.warn(msg, RuntimeWarning, stacklevel=2) 

521 

522 return svmem(total, avail, percent, used, free, 

523 active, inactive, buffers, cached, shared, slab)

526def swap_memory(): 

527 """Return swap memory metrics.""" 

528 mems = {} 

529 with open_binary('%s/meminfo' % get_procfs_path()) as f: 

530 for line in f: 

531 fields = line.split() 

532 mems[fields[0]] = int(fields[1]) * 1024 

533 # We prefer /proc/meminfo over sysinfo() syscall so that 

534 # psutil.PROCFS_PATH can be used in order to allow retrieval 

535 # for linux containers, see: 

536 # https://github.com/giampaolo/psutil/issues/1015 

537 try: 

538 total = mems[b'SwapTotal:'] 

539 free = mems[b'SwapFree:'] 

540 except KeyError: 

541 _, _, _, _, total, free, unit_multiplier = cext.linux_sysinfo() 

542 total *= unit_multiplier 

543 free *= unit_multiplier 

544 

545 used = total - free 

546 percent = usage_percent(used, total, round_=1) 

547 # get pgin/pgouts 

548 try: 

549 f = open_binary("%s/vmstat" % get_procfs_path()) 

550 except IOError as err: 

551 # see https://github.com/giampaolo/psutil/issues/722 

552 msg = "'sin' and 'sout' swap memory stats couldn't " + \ 

553 "be determined and were set to 0 (%s)" % str(err) 

554 warnings.warn(msg, RuntimeWarning, stacklevel=2) 

555 sin = sout = 0 

556 else: 

557 with f: 

558 sin = sout = None 

559 for line in f: 

560 # values are expressed in 4 kilo bytes, we want 

561 # bytes instead 

562 if line.startswith(b'pswpin'): 

563 sin = int(line.split(b' ')[1]) * 4 * 1024 

564 elif line.startswith(b'pswpout'): 

565 sout = int(line.split(b' ')[1]) * 4 * 1024 

566 if sin is not None and sout is not None: 

567 break 

568 else: 

569 # we might get here when dealing with exotic Linux 

570 # flavors, see: 

571 # https://github.com/giampaolo/psutil/issues/313 

572 msg = "'sin' and 'sout' swap memory stats couldn't " 

573 msg += "be determined and were set to 0" 

574 warnings.warn(msg, RuntimeWarning, stacklevel=2) 

575 sin = sout = 0 

576 return _common.sswap(total, used, free, percent, sin, sout)

584def cpu_times(): 

585 """Return a named tuple representing the following system-wide 

586 CPU times: 

587 (user, nice, system, idle, iowait, irq, softirq [steal, [guest, 

588 [guest_nice]]]) 

589 Last 3 fields may not be available on all Linux kernel versions.

654 # */thread_siblings_list is deprecated and may disappear in the future. 

655 # https://www.kernel.org/doc/Documentation/admin-guide/cputopology.rst

669 current_info = {}

671 for line in f:

776class Connections: 

777 """A wrapper on top of /proc/net/* files, retrieving per-process 

778 and system-wide open connections (TCP, UDP, UNIX) similarly to 

779 "netstat -an". 

780 

781 Note: in case of UNIX sockets we're only able to determine the 

782 local endpoint/path, not the one it's connected to. 

783 According to [1] it would be possible but not easily. 

784 

785 [1] http://serverfault.com/a/417946 

786 """ 

787 

788 def __init__(self): 

789 # The string represents the basename of the corresponding 

790 # /proc/net/{proto_name} file. 

791 tcp4 = ("tcp", socket.AF_INET, socket.SOCK_STREAM) 

792 tcp6 = ("tcp6", socket.AF_INET6, socket.SOCK_STREAM) 

793 udp4 = ("udp", socket.AF_INET, socket.SOCK_DGRAM) 

794 udp6 = ("udp6", socket.AF_INET6, socket.SOCK_DGRAM) 

795 unix = ("unix", socket.AF_UNIX, None) 

796 self.tmap = { 

797 "all": (tcp4, tcp6, udp4, udp6, unix), 

798 "tcp": (tcp4, tcp6), 

799 "tcp4": (tcp4,), 

800 "tcp6": (tcp6,), 

801 "udp": (udp4, udp6), 

802 "udp4": (udp4,), 

803 "udp6": (udp6,), 

804 "unix": (unix,), 

805 "inet": (tcp4, tcp6, udp4, udp6), 

806 "inet4": (tcp4, udp4), 

807 "inet6": (tcp6, udp6), 

808 } 

809 self._procfs_path = None

811 def get_proc_inodes(self, pid): 

812 inodes = defaultdict(list) 

813 for fd in os.listdir("%s/%s/fd" % (self._procfs_path, pid)): 

814 try: 

815 inode = readlink("%s/%s/fd/%s" % (self._procfs_path, pid, fd)) 

816 except (FileNotFoundError, ProcessLookupError): 

817 # ENOENT == file which is gone in the meantime; 

818 # os.stat('/proc/%s' % self.pid) will be done later 

819 # to force NSP (if it's the case) 

820 continue 

821 except OSError as err: 

822 if err.errno == errno.EINVAL: 

823 # not a link 

824 continue 

825 if err.errno == errno.ENAMETOOLONG: 

826 # file name too long 

827 debug(err) 

828 continue 

829 raise 

830 else: 

831 if inode.startswith('socket:['): 

832 # the process is using a socket 

833 inode = inode[8:][:-1] 

834 inodes[inode].append((pid, int(fd))) 

835 return inodes

837 def get_all_inodes(self): 

838 inodes = {} 

839 for pid in pids(): 

840 try: 

841 inodes.update(self.get_proc_inodes(pid)) 

842 except (FileNotFoundError, ProcessLookupError, PermissionError): 

843 # os.listdir() is gonna raise a lot of access denied 

844 # exceptions in case of unprivileged user; that's fine 

845 # as we'll just end up returning a connection with PID 

846 # and fd set to None anyway. 

847 # Both netstat -an and lsof does the same so it's 

848 # unlikely we can do any better. 

849 # ENOENT just means a PID disappeared on us. 

850 continue 

851 return inodes

854 def decode_address(addr, family): 

855 """Accept an "ip:port" address as displayed in /proc/net/* 

856 and convert it into a human readable form, like: 

857 

858 "0500000A:0016" -> ("10.0.0.5", 22) 

859 "0000000000000000FFFF00000100007F:9E49" -> ("::ffff:127.0.0.1", 40521) 

860 

861 The IP address portion is a little or big endian four-byte 

862 hexadecimal number; that is, the least significant byte is listed 

863 first, so we need to reverse the order of the bytes to convert it 

864 to an IP address. 

865 The port is represented as a two-byte hexadecimal number. 

866 

867 Reference: 

868 http://linuxdevcenter.com/pub/a/linux/2000/11/16/LinuxAdmin.html 

869 """ 

870 ip, port = addr.split(':') 

871 port = int(port, 16) 

872 # this usually refers to a local socket in listen mode with 

873 # no end-points connected 

874 if not port: 

875 return () 

876 if PY3: 

877 ip = ip.encode('ascii') 

878 if family == socket.AF_INET: 

879 # see: https://github.com/giampaolo/psutil/issues/201 

880 if LITTLE_ENDIAN: 

881 ip = socket.inet_ntop(family, base64.b16decode(ip)[::-1]) 

882 else: 

883 ip = socket.inet_ntop(family, base64.b16decode(ip)) 

884 else: # IPv6 

885 ip = base64.b16decode(ip) 

886 try: 

887 # see: https://github.com/giampaolo/psutil/issues/201 

888 if LITTLE_ENDIAN: 

889 ip = socket.inet_ntop( 

890 socket.AF_INET6, 

891 struct.pack('>4I', *struct.unpack('<4I', ip))) 

892 else: 

893 ip = socket.inet_ntop( 

894 socket.AF_INET6, 

895 struct.pack('<4I', *struct.unpack('<4I', ip))) 

896 except ValueError: 

897 # see: https://github.com/giampaolo/psutil/issues/623 

898 if not supports_ipv6(): 

899 raise _Ipv6UnsupportedError 

900 else: 

901 raise 

902 return _common.addr(ip, port)

905 def process_inet(file, family, type_, inodes, filter_pid=None): 

906 """Parse /proc/net/tcp* and /proc/net/udp* files.""" 

907 if file.endswith('6') and not os.path.exists(file): 

908 # IPv6 not supported 

909 return 

910 with open_text(file) as f: 

911 f.readline() # skip the first line 

912 for lineno, line in enumerate(f, 1): 

913 try: 

914 _, laddr, raddr, status, _, _, _, _, _, inode = \ 

915 line.split()[:10] 

916 except ValueError: 

917 raise RuntimeError( 

918 "error while parsing %s; malformed line %s %r" % ( 

919 file, lineno, line)) 

920 if inode in inodes: 

921 # # We assume inet sockets are unique, so we error 

922 # # out if there are multiple references to the 

923 # # same inode. We won't do this for UNIX sockets. 

924 # if len(inodes[inode]) > 1 and family != socket.AF_UNIX: 

925 # raise ValueError("ambiguous inode with multiple " 

926 # "PIDs references") 

927 pid, fd = inodes[inode][0] 

928 else: 

929 pid, fd = None, -1 

930 if filter_pid is not None and filter_pid != pid: 

931 continue 

932 else: 

933 if type_ == socket.SOCK_STREAM: 

934 status = TCP_STATUSES[status] 

935 else: 

936 status = _common.CONN_NONE 

937 try: 

938 laddr = Connections.decode_address(laddr, family) 

939 raddr = Connections.decode_address(raddr, family) 

940 except _Ipv6UnsupportedError: 

941 continue 

942 yield (fd, family, type_, laddr, raddr, status, pid)

944 @staticmethod 

945 def process_unix(file, family, inodes, filter_pid=None): 

946 """Parse /proc/net/unix files.""" 

947 with open_text(file) as f: 

948 f.readline() # skip the first line 

949 for line in f: 

950 tokens = line.split() 

951 try: 

952 _, _, _, _, type_, _, inode = tokens[0:7] 

953 except ValueError: 

954 if ' ' not in line: 

955 # see: https://github.com/giampaolo/psutil/issues/766 

956 continue 

957 raise RuntimeError( 

958 "error while parsing %s; malformed line %r" % ( 

959 file, line)) 

960 if inode in inodes: # noqa 

961 # With UNIX sockets we can have a single inode 

962 # referencing many file descriptors. 

963 pairs = inodes[inode] 

964 else: 

965 pairs = [(None, -1)] 

966 for pid, fd in pairs: 

967 if filter_pid is not None and filter_pid != pid: 

968 continue 

969 else: 

970 path = tokens[-1] if len(tokens) == 8 else '' 

971 type_ = _common.socktype_to_enum(int(type_)) 

972 # XXX: determining the remote endpoint of a 

973 # UNIX socket on Linux is not possible, see: 

974 # https://serverfault.com/questions/252723/ 

975 raddr = "" 

976 status = _common.CONN_NONE 

977 yield (fd, family, type_, path, raddr, status, pid)

979 def retrieve(self, kind, pid=None): 

980 if kind not in self.tmap: 

981 raise ValueError("invalid %r kind argument; choose between %s" 

982 % (kind, ', '.join([repr(x) for x in self.tmap]))) 

983 self._procfs_path = get_procfs_path() 

984 if pid is not None: 

985 inodes = self.get_proc_inodes(pid) 

986 if not inodes: 

987 # no connections for this process 

988 return [] 

989 else: 

990 inodes = self.get_all_inodes() 

991 ret = set() 

992 for proto_name, family, type_ in self.tmap[kind]: 

993 path = "%s/net/%s" % (self._procfs_path, proto_name) 

994 if family in (socket.AF_INET, socket.AF_INET6): 

995 ls = self.process_inet( 

996 path, family, type_, inodes, filter_pid=pid) 

997 else: 

998 ls = self.process_unix( 

999 path, family, inodes, filter_pid=pid) 

1000 for fd, family, type_, laddr, raddr, status, bound_pid in ls: 

1001 if pid: 

1002 conn = _common.pconn(fd, family, type_, laddr, raddr, 

1003 status) 

1004 else: 

1005 conn = _common.sconn(fd, family, type_, laddr, raddr, 

1006 status, bound_pid) 

1007 ret.add(conn) 

1008 return list(ret)

1090def disk_io_counters(perdisk=False): 

1091 """Return disk I/O statistics for every disk installed on the 

1092 system as a dict of raw tuples. 

1093 """ 

1094 def read_procfs(): 

1095 # OK, this is a bit confusing. The format of /proc/diskstats can 

1096 # have 3 variations. 

1097 # On Linux 2.4 each line has always 15 fields, e.g.: 

1098 # "3 0 8 hda 8 8 8 8 8 8 8 8 8 8 8" 

1099 # On Linux 2.6+ each line *usually* has 14 fields, and the disk 

1100 # name is in another position, like this: 

1101 # "3 0 hda 8 8 8 8 8 8 8 8 8 8 8" 

1102 # ...unless (Linux 2.6) the line refers to a partition instead 

1103 # of a disk, in which case the line has less fields (7): 

1104 # "3 1 hda1 8 8 8 8" 

1105 # 4.18+ has 4 fields added: 

1106 # "3 0 hda 8 8 8 8 8 8 8 8 8 8 8 0 0 0 0" 

1107 # 5.5 has 2 more fields. 

1108 # See: 

1109 # https://www.kernel.org/doc/Documentation/iostats.txt 

1110 # https://www.kernel.org/doc/Documentation/ABI/testing/procfs-diskstats 

1111 with open_text("%s/diskstats" % get_procfs_path()) as f: 

1112 lines = f.readlines() 

1113 for line in lines: 

1114 fields = line.split() 

1115 flen = len(fields) 

1116 if flen == 15: 

1117 # Linux 2.4 

1118 name = fields[3] 

1119 reads = int(fields[2]) 

1120 (reads_merged, rbytes, rtime, writes, writes_merged, 

1121 wbytes, wtime, _, busy_time, _) = map(int, fields[4:14]) 

1122 elif flen == 14 or flen >= 18: 

1123 # Linux 2.6+, line referring to a disk 

1124 name = fields[2] 

1125 (reads, reads_merged, rbytes, rtime, writes, writes_merged, 

1126 wbytes, wtime, _, busy_time, _) = map(int, fields[3:14]) 

1127 elif flen == 7: 

1128 # Linux 2.6+, line referring to a partition 

1129 name = fields[2] 

1130 reads, rbytes, writes, wbytes = map(int, fields[3:]) 

1131 rtime = wtime = reads_merged = writes_merged = busy_time = 0 

1132 else: 

1133 raise ValueError("not sure how to interpret line %r" % line) 

1134 yield (name, reads, writes, rbytes, wbytes, rtime, wtime, 

1135 reads_merged, writes_merged, busy_time)

1137 def read_sysfs(): 

1138 for block in os.listdir('/sys/block'): 

1139 for root, _, files in os.walk(os.path.join('/sys/block', block)): 

1140 if 'stat' not in files: 

1141 continue

1143 fields = f.read().strip().split() 

1144 name = os.path.basename(root) 

1145 (reads, reads_merged, rbytes, rtime, writes, writes_merged, 

1146 wbytes, wtime, _, busy_time) = map(int, fields[:10])

1148 wtime, reads_merged, writes_merged, busy_time) 

1149 

1150 if os.path.exists('%s/diskstats' % get_procfs_path()): 

1151 gen = read_procfs() 

1152 elif os.path.exists('/sys/block'): 

1153 gen = read_sysfs()

1157 "system" % get_procfs_path())

1160 for entry in gen: 

1161 (name, reads, writes, rbytes, wbytes, rtime, wtime, reads_merged, 

1162 writes_merged, busy_time) = entry 

1163 if not perdisk and not is_storage_device(name): 

1164 # perdisk=False means we want to calculate totals so we skip 

1165 # partitions (e.g. 'sda1', 'nvme0n1p1') and only include 

1166 # base disk devices (e.g. 'sda', 'nvme0n1'). Base disks 

1167 # include a total of all their partitions + some extra size 

1168 # of their own: 

1169 # $ cat /proc/diskstats 

1170 # 259 0 sda 10485760 ... 

1171 # 259 1 sda1 5186039 ... 

1172 # 259 1 sda2 5082039 ... 

1173 # See: 

1174 # https://github.com/giampaolo/psutil/pull/1313 

1175 continue 

1176 

1177 rbytes *= DISK_SECTOR_SIZE 

1178 wbytes *= DISK_SECTOR_SIZE 

1179 retdict[name] = (reads, writes, rbytes, wbytes, rtime, wtime, 

1180 reads_merged, writes_merged, busy_time) 

1181 

1182 return retdict

1200 def ask_proc_partitions(self):

1207 minor = int(fields[1]) if fields[1].isdigit() else None

1210 if name: # just for extra safety

1232 data = f.read().strip()

1306def sensors_temperatures():

1313 retrieve this info, and this implementation relies on it 

1314 only (old distros will probably use something else) 

1315 - lm-sensors on Ubuntu 16.04 relies on /sys/class/hwmon 

1316 - /sys/class/thermal/thermal_zone* is another one but it's more

1343 path = os.path.join(os.path.dirname(base), 'name') 

1344 unit_name = cat(path).strip() 

1345 except (IOError, OSError, ValueError):

1347 # whole entry. Sure thing is Linux's /sys/class/hwmon really 

1348 # is a stinky broken mess. 

1349 # https://github.com/giampaolo/psutil/issues/1009 

1350 # https://github.com/giampaolo/psutil/issues/1101 

1351 # https://github.com/giampaolo/psutil/issues/1129 

1352 # https://github.com/giampaolo/psutil/issues/1245 

1353 # https://github.com/giampaolo/psutil/issues/1323

1357 critical = bcat(base + '_crit', fallback=None)

1359 

1360 if high is not None: 

1361 try: 

1362 high = float(high) / 1000.0 

1363 except ValueError: 

1364 high = None 

1365 if critical is not None: 

1366 try: 

1367 critical = float(critical) / 1000.0 

1368 except ValueError: 

1369 critical = None

1372 

1373 # Indication that no sensors were detected in /sys/class/hwmon/ 

1374 if not basenames:

1376 basenames = sorted(set(basenames))

1378 for base in basenames: 

1379 try: 

1380 path = os.path.join(base, 'temp') 

1381 current = float(bcat(path)) / 1000.0 

1382 path = os.path.join(base, 'type') 

1383 unit_name = cat(path).strip() 

1384 except (IOError, OSError, ValueError) as err: 

1385 debug(err) 

1386 continue

1388 trip_paths = glob.glob(base + '/trip_point*') 

1389 trip_points = set(['_'.join( 

1390 os.path.basename(p).split('_')[0:3]) for p in trip_paths]) 

1391 critical = None 

1392 high = None

1401 fallback=None) 

1402 

1403 if high is not None: 

1404 try: 

1405 high = float(high) / 1000.0 

1406 except ValueError: 

1407 high = None 

1408 if critical is not None: 

1409 try: 

1410 critical = float(critical) / 1000.0 

1411 except ValueError: 

1412 critical = None

1450def sensors_battery(): 

1451 """Return battery information. 

1452 Implementation note: it appears /sys/class/power_supply/BAT0/ 

1453 directory structure may vary and provide files with the same 

1454 meaning but under different names, see: 

1455 https://github.com/giampaolo/psutil/issues/966.

1459 def multi_bcat(*paths): 

1460 """Attempt to read the content of multiple files which may 

1461 not exist. If none of them exist return None. 

1462 """ 

1463 for path in paths: 

1464 ret = bcat(path, fallback=null) 

1465 if ret != null: 

1466 try: 

1467 return int(ret) 

1468 except ValueError: 

1469 return ret.strip() 

1470 return None 

1471 

1472 bats = [x for x in os.listdir(POWER_SUPPLY_PATH) if x.startswith('BAT') or 

1473 'battery' in x.lower()]

1477 # some rare exceptions: 

1478 # https://github.com/giampaolo/psutil/issues/1238 

1479 root = os.path.join(POWER_SUPPLY_PATH, sorted(bats)[0]) 

1480 

1481 # Base metrics. 

1482 energy_now = multi_bcat( 

1483 root + "/energy_now", 

1484 root + "/charge_now") 

1485 power_now = multi_bcat( 

1486 root + "/power_now", 

1487 root + "/current_now") 

1488 energy_full = multi_bcat( 

1489 root + "/energy_full", 

1490 root + "/charge_full") 

1491 time_to_empty = multi_bcat(root + "/time_to_empty_now") 

1492 

1493 # Percent. If we have energy_full the percentage will be more 

1494 # accurate compared to reading /capacity file (float vs. int). 

1495 if energy_full is not None and energy_now is not None: 

1496 try: 

1497 percent = 100.0 * energy_now / energy_full 

1498 except ZeroDivisionError: 

1499 percent = 0.0 

1500 else: 

1501 percent = int(cat(root + "/capacity", fallback=-1)) 

1502 if percent == -1: 

1503 return None 

1504 

1505 # Is AC power cable plugged in? 

1506 # Note: AC0 is not always available and sometimes (e.g. CentOS7) 

1507 # it's called "AC". 

1508 power_plugged = None 

1509 online = multi_bcat( 

1510 os.path.join(POWER_SUPPLY_PATH, "AC0/online"), 

1511 os.path.join(POWER_SUPPLY_PATH, "AC/online")) 

1512 if online is not None: 

1513 power_plugged = online == 1 

1514 else: 

1515 status = cat(root + "/status", fallback="").strip().lower() 

1516 if status == "discharging": 

1517 power_plugged = False 

1518 elif status in ("charging", "full"): 

1519 power_plugged = True 

1520 

1521 # Seconds left. 

1522 # Note to self: we may also calculate the charging ETA as per: 

1523 # https://github.com/thialfihar/dotfiles/blob/ 

1524 # 013937745fd9050c30146290e8f963d65c0179e6/bin/battery.py#L55 

1525 if power_plugged: 

1526 secsleft = _common.POWER_TIME_UNLIMITED 

1527 elif energy_now is not None and power_now is not None: 

1528 try: 

1529 secsleft = int(energy_now / power_now * 3600) 

1530 except ZeroDivisionError: 

1531 secsleft = _common.POWER_TIME_UNKNOWN 

1532 elif time_to_empty is not None: 

1533 secsleft = int(time_to_empty * 60) 

1534 if secsleft < 0: 

1535 secsleft = _common.POWER_TIME_UNKNOWN 

1536 else: 

1537 secsleft = _common.POWER_TIME_UNKNOWN

1539 return _common.sbattery(percent, secsleft, power_plugged)

1589def pid_exists(pid): 

1590 """Check for the existence of a unix PID. Linux TIDs are not 

1591 supported (always return False). 

1592 """ 

1593 if not _psposix.pid_exists(pid): 

1594 return False 

1595 else: 

1596 # Linux's apparently does not distinguish between PIDs and TIDs 

1597 # (thread IDs). 

1598 # listdir("/proc") won't show any TID (only PIDs) but 

1599 # os.stat("/proc/{tid}") will succeed if {tid} exists. 

1600 # os.kill() can also be passed a TID. This is quite confusing. 

1601 # In here we want to enforce this distinction and support PIDs 

1602 # only, see: 

1603 # https://github.com/giampaolo/psutil/issues/687 

1604 try: 

1605 # Note: already checked that this is faster than using a 

1606 # regular expr. Also (a lot) faster than doing 

1607 # 'return pid in pids()' 

1608 path = "%s/%s/status" % (get_procfs_path(), pid) 

1609 with open_binary(path) as f: 

1610 for line in f: 

1611 if line.startswith(b"Tgid:"): 

1612 tgid = int(line.split()[1]) 

1613 # If tgid and pid are the same then we're 

1614 # dealing with a process PID. 

1615 return tgid == pid 

1616 raise ValueError("'Tgid' line not found in %s" % path) 

1617 except (EnvironmentError, ValueError): 

1618 return pid in pids()

1643def wrap_exceptions(fun): 

1644 """Decorator which translates bare OSError and IOError exceptions 

1645 into NoSuchProcess and AccessDenied. 

1646 """ 

1647 @functools.wraps(fun) 

1648 def wrapper(self, *args, **kwargs): 

1649 try: 

1650 return fun(self, *args, **kwargs) 

1651 except PermissionError: 

1652 raise AccessDenied(self.pid, self._name) 

1653 except ProcessLookupError: 

1654 self._raise_if_zombie() 

1655 raise NoSuchProcess(self.pid, self._name) 

1656 except FileNotFoundError: 

1657 self._raise_if_zombie() 

1658 if not os.path.exists("%s/%s" % (self._procfs_path, self.pid)): 

1659 raise NoSuchProcess(self.pid, self._name) 

1660 raise 

1661 return wrapper

1675 def _is_zombie(self): 

1676 # Note: most of the times Linux is able to return info about the 

1677 # process even if it's a zombie, and /proc/{pid} will exist. 

1678 # There are some exceptions though, like exe(), cmdline() and 

1679 # memory_maps(). In these cases /proc/{pid}/{file} exists but 

1680 # it's empty. Instead of returning a "null" value we'll raise an 

1681 # exception. 

1682 try: 

1683 data = bcat("%s/%s/stat" % (self._procfs_path, self.pid)) 

1684 except (IOError, OSError): 

1685 return False 

1686 else: 

1687 rpar = data.rfind(b')') 

1688 status = data[rpar + 2:rpar + 3] 

1689 return status == b"Z"

1691 def _raise_if_zombie(self): 

1692 if self._is_zombie(): 

1693 raise ZombieProcess(self.pid, self._name, self._ppid)

1696 """Raise NSP if the process disappeared on us."""

1698 # incorrect or incomplete result. 

1699 os.stat('%s/%s' % (self._procfs_path, self.pid))

1703 def _parse_stat_file(self): 

1704 """Parse /proc/{pid}/stat file and return a dict with various 

1705 process info. 

1706 Using "man proc" as a reference: where "man proc" refers to 

1707 position N always subtract 3 (e.g ppid position 4 in 

1708 'man proc' == position 1 in here). 

1709 The return value is cached in case oneshot() ctx manager is 

1710 in use. 

1711 """ 

1712 data = bcat("%s/%s/stat" % (self._procfs_path, self.pid)) 

1713 # Process name is between parentheses. It can contain spaces and 

1714 # other parentheses. This is taken into account by looking for 

1715 # the first occurrence of "(" and the last occurrence of ")". 

1716 rpar = data.rfind(b')') 

1717 name = data[data.find(b'(') + 1:rpar] 

1718 fields = data[rpar + 2:].split() 

1719 

1720 ret = {} 

1721 ret['name'] = name 

1722 ret['status'] = fields[0] 

1723 ret['ppid'] = fields[1] 

1724 ret['ttynr'] = fields[4] 

1725 ret['utime'] = fields[11] 

1726 ret['stime'] = fields[12] 

1727 ret['children_utime'] = fields[13] 

1728 ret['children_stime'] = fields[14] 

1729 ret['create_time'] = fields[19] 

1730 ret['cpu_num'] = fields[36] 

1731 ret['blkio_ticks'] = fields[39] # aka 'delayacct_blkio_ticks' 

1732 

1733 return ret

1906 def _parse_smaps_rollup(self): 

1907 # /proc/pid/smaps_rollup was added to Linux in 2017. Faster 

1908 # than /proc/pid/smaps. It reports higher PSS than */smaps 

1909 # (from 1k up to 200k higher; tested against all processes). 

1910 # IMPORTANT: /proc/pid/smaps_rollup is weird, because it 

1911 # raises ESRCH / ENOENT for many PIDs, even if they're alive 

1912 # (also as root). In that case we'll use /proc/pid/smaps as 

1913 # fallback, which is slower but has a +50% success rate 

1914 # compared to /proc/pid/smaps_rollup. 

1915 uss = pss = swap = 0 

1916 with open_binary("{}/{}/smaps_rollup".format( 

1917 self._procfs_path, self.pid)) as f: 

1918 for line in f: 

1919 if line.startswith(b"Private_"): 

1920 # Private_Clean, Private_Dirty, Private_Hugetlb 

1921 uss += int(line.split()[1]) * 1024 

1922 elif line.startswith(b"Pss:"): 

1923 pss = int(line.split()[1]) * 1024 

1924 elif line.startswith(b"Swap:"): 

1925 swap = int(line.split()[1]) * 1024 

1926 return (uss, pss, swap)

1928 @wrap_exceptions 

1929 def _parse_smaps( 

1930 self, 

1931 # Gets Private_Clean, Private_Dirty, Private_Hugetlb. 

1932 _private_re=re.compile(br"\nPrivate.*:\s+(\d+)"), 

1933 _pss_re=re.compile(br"\nPss\:\s+(\d+)"), 

1934 _swap_re=re.compile(br"\nSwap\:\s+(\d+)")): 

1935 # /proc/pid/smaps does not exist on kernels < 2.6.14 or if 

1936 # CONFIG_MMU kernel configuration option is not enabled. 

1937 

1938 # Note: using 3 regexes is faster than reading the file 

1939 # line by line. 

1940 # XXX: on Python 3 the 2 regexes are 30% slower than on 

1941 # Python 2 though. Figure out why. 

1942 # 

1943 # You might be tempted to calculate USS by subtracting 

1944 # the "shared" value from the "resident" value in 

1945 # /proc/<pid>/statm. But at least on Linux, statm's "shared" 

1946 # value actually counts pages backed by files, which has 

1947 # little to do with whether the pages are actually shared. 

1948 # /proc/self/smaps on the other hand appears to give us the 

1949 # correct information. 

1950 smaps_data = self._read_smaps_file() 

1951 # Note: smaps file can be empty for certain processes. 

1952 # The code below will not crash though and will result to 0. 

1953 uss = sum(map(int, _private_re.findall(smaps_data))) * 1024 

1954 pss = sum(map(int, _pss_re.findall(smaps_data))) * 1024 

1955 swap = sum(map(int, _swap_re.findall(smaps_data))) * 1024 

1956 return (uss, pss, swap)
