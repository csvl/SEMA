17# LineProgramEntry - an entry in the line program. 

18# A line program is a sequence of encoded entries. Some of these entries add a 

19# new LineState (mapping between line and address), and some don't. 

20# 

21# command: 

22# The command/opcode - always numeric. For standard commands - it's the opcode 

23# that can be matched with one of the DW_LNS_* constants. For extended commands 

24# it's the extended opcode that can be matched with one of the DW_LNE_* 

25# constants. For special commands, it's the opcode itself. 

26# 

27# args: 

28# A list of decoded arguments of the command. 

29# 

30# is_extended: 

31# Since extended commands are encoded by a zero followed by an extended 

32# opcode, and these extended opcodes overlap with other opcodes, this 

33# flag is needed to mark that the command has an extended opcode. 

34# 

35# state: 

36# For commands that add a new state, it's the relevant LineState object. 

37# For commands that don't add a new state, it's None. 

38# 

39LineProgramEntry = namedtuple( 

40 'LineProgramEntry', 'command is_extended args state')

43class LineState(object): 

44 """ Represents a line program state (or a "row" in the matrix 

45 describing debug location information for addresses). 

46 The instance variables of this class are the "state machine registers" 

47 described in section 6.2.2 of DWARFv3 

48 """ 

49 def __init__(self, default_is_stmt): 

50 self.address = 0 

51 self.file = 1 

52 self.line = 1 

53 self.column = 0 

54 self.op_index = 0 

55 self.is_stmt = default_is_stmt 

56 self.basic_block = False 

57 self.end_sequence = False 

58 self.prologue_end = False 

59 self.epilogue_begin = False 

60 self.isa = 0 

61 self.discriminator = 0 

62 

63 def __repr__(self): 

64 a = ['<LineState %x:' % id(self)] 

65 a.append(' address = 0x%x' % self.address) 

66 for attr in ('file', 'line', 'column', 'is_stmt', 'basic_block', 

67 'end_sequence', 'prologue_end', 'epilogue_begin', 'isa', 

68 'discriminator'): 

69 a.append(' %s = %s' % (attr, getattr(self, attr))) 

70 return '\n'.join(a) + '>\n'

73class LineProgram(object): 

74 """ Builds a "line table", which is essentially the matrix described 

75 in section 6.2 of DWARFv3. It's a list of LineState objects, 

76 sorted by increasing address, so it can be used to obtain the 

77 state information for each address. 

78 """ 

79 def __init__(self, header, stream, structs, 

80 program_start_offset, program_end_offset): 

81 """ 

82 header: 

83 The header of this line program. Note: LineProgram may modify 

84 its header by appending file entries if DW_LNE_define_file 

85 instructions are encountered. 

86 

87 stream: 

88 The stream this program can be read from. 

89 

90 structs: 

91 A DWARFStructs instance suitable for this line program 

92 

93 program_{start|end}_offset: 

94 Offset in the debug_line section stream where this program 

95 starts (the actual program, after the header), and where it 

96 ends. 

97 The actual range includes start but not end: [start, end - 1] 

98 """ 

99 self.stream = stream 

100 self.header = header 

101 self.structs = structs 

102 self.program_start_offset = program_start_offset 

103 self.program_end_offset = program_end_offset 

104 self._decoded_entries = None 

105 

106 def get_entries(self): 

107 """ Get the decoded entries for this line program. Return a list of 

108 LineProgramEntry objects. 

109 Note that this contains more information than absolutely required 

110 for the line table. The line table can be easily extracted from 

111 the list of entries by looking only at entries with non-None 

112 state. The extra information is mainly for the purposes of display 

113 with readelf and debugging. 

114 """ 

115 if self._decoded_entries is None: 

116 self._decoded_entries = self._decode_line_program() 

117 return self._decoded_entries 

118 

119 #------ PRIVATE ------# 

120 

121 def __getitem__(self, name): 

122 """ Implement dict-like access to header entries 

123 """ 

124 return self.header[name]

126 def _decode_line_program(self): 

127 entries = [] 

128 state = LineState(self.header['default_is_stmt']) 

129 

130 def add_entry_new_state(cmd, args, is_extended=False): 

131 # Add an entry that sets a new state. 

132 # After adding, clear some state registers. 

133 entries.append(LineProgramEntry( 

134 cmd, is_extended, args, copy.copy(state))) 

135 state.discriminator = 0 

136 state.basic_block = False 

137 state.prologue_end = False 

138 state.epilogue_begin = False 

139 

140 def add_entry_old_state(cmd, args, is_extended=False): 

141 # Add an entry that doesn't visibly set a new state 

142 entries.append(LineProgramEntry(cmd, is_extended, args, None)) 

143 

144 offset = self.program_start_offset 

145 while offset < self.program_end_offset: 

146 opcode = struct_parse( 

147 self.structs.Dwarf_uint8(''), 

148 self.stream, 

149 offset) 

150 

151 # As an exercise in avoiding premature optimization, if...elif 

152 # chains are used here for standard and extended opcodes instead 

153 # of dispatch tables. This keeps the code much cleaner. Besides, 

154 # the majority of instructions in a typical program are special 

155 # opcodes anyway. 

156 if opcode >= self.header['opcode_base']: 

157 # Special opcode (follow the recipe in 6.2.5.1) 

158 maximum_operations_per_instruction = self['maximum_operations_per_instruction'] 

159 adjusted_opcode = opcode - self['opcode_base'] 

160 operation_advance = adjusted_opcode // self['line_range'] 

161 address_addend = ( 

162 self['minimum_instruction_length'] * 

163 ((state.op_index + operation_advance) // 

164 maximum_operations_per_instruction)) 

165 state.address += address_addend 

166 state.op_index = (state.op_index + operation_advance) % maximum_operations_per_instruction 

167 line_addend = self['line_base'] + (adjusted_opcode % self['line_range']) 

168 state.line += line_addend 

169 add_entry_new_state( 

170 opcode, [line_addend, address_addend, state.op_index]) 

171 elif opcode == 0: 

172 # Extended opcode: start with a zero byte, followed by 

173 # instruction size and the instruction itself. 

174 inst_len = struct_parse(self.structs.Dwarf_uleb128(''), 

175 self.stream) 

176 ex_opcode = struct_parse(self.structs.Dwarf_uint8(''), 

177 self.stream) 

178 

179 if ex_opcode == DW_LNE_end_sequence: 

180 state.end_sequence = True 

181 state.is_stmt = 0 

182 add_entry_new_state(ex_opcode, [], is_extended=True) 

183 # reset state 

184 state = LineState(self.header['default_is_stmt']) 

185 elif ex_opcode == DW_LNE_set_address: 

186 operand = struct_parse(self.structs.Dwarf_target_addr(''), 

187 self.stream) 

188 state.address = operand 

189 add_entry_old_state(ex_opcode, [operand], is_extended=True) 

190 elif ex_opcode == DW_LNE_define_file: 

191 operand = struct_parse( 

192 self.structs.Dwarf_lineprog_file_entry, self.stream) 

193 self['file_entry'].append(operand) 

194 add_entry_old_state(ex_opcode, [operand], is_extended=True) 

195 elif ex_opcode == DW_LNE_set_discriminator: 

196 operand = struct_parse(self.structs.Dwarf_uleb128(''), 

197 self.stream) 

198 state.discriminator = operand 

199 else: 

200 # Unknown, but need to roll forward the stream because the 

201 # length is specified. Seek forward inst_len - 1 because 

202 # we've already read the extended opcode, which takes part 

203 # in the length. 

204 self.stream.seek(inst_len - 1, os.SEEK_CUR) 

205 else: # 0 < opcode < opcode_base 

206 # Standard opcode 

207 if opcode == DW_LNS_copy: 

208 add_entry_new_state(opcode, []) 

209 elif opcode == DW_LNS_advance_pc: 

210 operand = struct_parse(self.structs.Dwarf_uleb128(''), 

211 self.stream) 

212 address_addend = ( 

213 operand * self.header['minimum_instruction_length']) 

214 state.address += address_addend 

215 add_entry_old_state(opcode, [address_addend]) 

216 elif opcode == DW_LNS_advance_line: 

217 operand = struct_parse(self.structs.Dwarf_sleb128(''), 

218 self.stream) 

219 state.line += operand 

220 elif opcode == DW_LNS_set_file: 

221 operand = struct_parse(self.structs.Dwarf_uleb128(''), 

222 self.stream) 

223 state.file = operand 

224 add_entry_old_state(opcode, [operand]) 

225 elif opcode == DW_LNS_set_column: 

226 operand = struct_parse(self.structs.Dwarf_uleb128(''), 

227 self.stream) 

228 state.column = operand 

229 add_entry_old_state(opcode, [operand]) 

230 elif opcode == DW_LNS_negate_stmt: 

231 state.is_stmt = not state.is_stmt 

232 add_entry_old_state(opcode, []) 

233 elif opcode == DW_LNS_set_basic_block: 

234 state.basic_block = True 

235 add_entry_old_state(opcode, []) 

236 elif opcode == DW_LNS_const_add_pc: 

237 adjusted_opcode = 255 - self['opcode_base'] 

238 address_addend = ((adjusted_opcode // self['line_range']) * 

239 self['minimum_instruction_length']) 

240 state.address += address_addend 

241 add_entry_old_state(opcode, [address_addend]) 

242 elif opcode == DW_LNS_fixed_advance_pc: 

243 operand = struct_parse(self.structs.Dwarf_uint16(''), 

244 self.stream) 

245 state.address += operand 

246 add_entry_old_state(opcode, [operand]) 

247 elif opcode == DW_LNS_set_prologue_end: 

248 state.prologue_end = True 

249 add_entry_old_state(opcode, []) 

250 elif opcode == DW_LNS_set_epilogue_begin: 

251 state.epilogue_begin = True 

252 add_entry_old_state(opcode, []) 

253 elif opcode == DW_LNS_set_isa: 

254 operand = struct_parse(self.structs.Dwarf_uleb128(''), 

255 self.stream) 

256 state.isa = operand 

257 add_entry_old_state(opcode, [operand]) 

258 else: 

259 dwarf_assert(False, 'Invalid standard line program opcode: %s' % ( 

260 opcode,)) 

261 offset = self.stream.tell() 

262 return entries
