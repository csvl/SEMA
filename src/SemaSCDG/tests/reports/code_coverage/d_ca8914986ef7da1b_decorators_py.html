25def not_implemented_for(*graph_types): 

26 """Decorator to mark algorithms as not implemented 

27 

28 Parameters 

29 ---------- 

30 graph_types : container of strings 

31 Entries must be one of "directed", "undirected", "multigraph", or "graph". 

32 

33 Returns 

34 ------- 

35 _require : function 

36 The decorated function. 

37 

38 Raises 

39 ------ 

40 NetworkXNotImplemented 

41 If any of the packages cannot be imported 

42 

43 Notes 

44 ----- 

45 Multiple types are joined logically with "and". 

46 For "or" use multiple @not_implemented_for() lines. 

47 

48 Examples 

49 -------- 

50 Decorate functions like this:: 

51 

52 @not_implemented_for("directed") 

53 def sp_function(G): 

54 pass 

55 

56 # rule out MultiDiGraph 

57 @not_implemented_for("directed","multigraph") 

58 def sp_np_function(G): 

59 pass 

60 

61 # rule out all except DiGraph 

62 @not_implemented_for("undirected") 

63 @not_implemented_for("multigraph") 

64 def sp_np_function(G): 

65 pass 

66 """ 

67 if ("directed" in graph_types) and ("undirected" in graph_types): 

68 raise ValueError("Function not implemented on directed AND undirected graphs?") 

69 if ("multigraph" in graph_types) and ("graph" in graph_types): 

70 raise ValueError("Function not implemented on graph AND multigraphs?") 

71 if not set(graph_types) < {"directed", "undirected", "multigraph", "graph"}: 

72 raise KeyError( 

73 "use one or more of directed, undirected, multigraph, graph. " 

74 f"You used {graph_types}" 

75 ) 

76 

77 # 3-way logic: True if "directed" input, False if "undirected" input, else None 

78 dval = ("directed" in graph_types) or "undirected" not in graph_types and None 

79 mval = ("multigraph" in graph_types) or "graph" not in graph_types and None 

80 errmsg = f"not implemented for {' '.join(graph_types)} type" 

81 

82 def _not_implemented_for(g): 

83 if (mval is None or mval == g.is_multigraph()) and ( 

84 dval is None or dval == g.is_directed() 

85 ): 

86 raise nx.NetworkXNotImplemented(errmsg) 

87 

88 return g 

89 

90 return argmap(_not_implemented_for, 0)

103def open_file(path_arg, mode="r"): 

104 """Decorator to ensure clean opening and closing of files. 

105 

106 Parameters 

107 ---------- 

108 path_arg : string or int 

109 Name or index of the argument that is a path. 

110 

111 mode : str 

112 String for opening mode. 

113 

114 Returns 

115 ------- 

116 _open_file : function 

117 Function which cleanly executes the io. 

118 

119 Examples 

120 -------- 

121 Decorate functions like this:: 

122 

123 @open_file(0,"r") 

124 def read_function(pathname): 

125 pass 

126 

127 @open_file(1,"w") 

128 def write_function(G, pathname): 

129 pass 

130 

131 @open_file(1,"w") 

132 def write_function(G, pathname="graph.dot"): 

133 pass 

134 

135 @open_file("pathname","w") 

136 def write_function(G, pathname="graph.dot"): 

137 pass 

138 

139 @open_file("path", "w+") 

140 def another_function(arg, **kwargs): 

141 path = kwargs["path"] 

142 pass 

143 

144 Notes 

145 ----- 

146 Note that this decorator solves the problem when a path argument is 

147 specified as a string, but it does not handle the situation when the 

148 function wants to accept a default of None (and then handle it). 

149 

150 Here is an example of how to handle this case:: 

151 

152 @open_file("path") 

153 def some_function(arg1, arg2, path=None): 

154 if path is None: 

155 fobj = tempfile.NamedTemporaryFile(delete=False) 

156 else: 

157 # `path` could have been a string or file object or something 

158 # similar. In any event, the decorator has given us a file object 

159 # and it will close it for us, if it should. 

160 fobj = path 

161 

162 try: 

163 fobj.write("blah") 

164 finally: 

165 if path is None: 

166 fobj.close() 

167 

168 Normally, we'd want to use "with" to ensure that fobj gets closed. 

169 However, the decorator will make `path` a file object for us, 

170 and using "with" would undesirably close that file object. 

171 Instead, we use a try block, as shown above. 

172 When we exit the function, fobj will be closed, if it should be, by the decorator. 

173 """ 

174 

175 def _open_file(path): 

176 # Now we have the path_arg. There are two types of input to consider: 

177 # 1) string representing a path that should be opened 

178 # 2) an already opened file object 

179 if isinstance(path, str): 

180 ext = splitext(path)[1] 

181 elif isinstance(path, Path): 

182 # path is a pathlib reference to a filename 

183 ext = path.suffix 

184 path = str(path) 

185 else: 

186 # could be None, or a file handle, in which case the algorithm will deal with it 

187 return path, lambda: None 

188 

189 fobj = _dispatch_dict[ext](path, mode=mode) 

190 return fobj, lambda: fobj.close() 

191 

192 return argmap(_open_file, path_arg, try_finally=True)

199 of nodes. If it is a number, the nodes used are `range(n)`. 

200 This allows `nx.complete_graph(50)` in place of `nx.complete_graph(list(range(50)))`. 

201 And it also allows `nx.complete_graph(any_list_of_nodes)`. 

202 

203 Parameters 

204 ---------- 

205 which_args : string or int or sequence of strings or ints 

206 If string, the name of the argument to be treated. 

207 If int, the index of the argument to be treated. 

208 If more than one node argument is allowed, can be a list of locations. 

209 

210 Returns 

211 ------- 

212 _nodes_or_numbers : function 

213 Function which replaces int args with ranges. 

214 

215 Examples 

216 -------- 

217 Decorate functions like this:: 

218 

219 @nodes_or_number("nodes") 

220 def empty_graph(nodes): 

221 # nodes is converted to a list of nodes 

222 

223 @nodes_or_number(0) 

224 def empty_graph(nodes): 

225 # nodes is converted to a list of nodes 

226

228 def grid_2d_graph(m1, m2, periodic=False): 

229 # m1 and m2 are each converted to a list of nodes 

230

232 def grid_2d_graph(m1, m2, periodic=False): 

233 # m1 and m2 are each converted to a list of nodes 

234 

235 @nodes_or_number(1) 

236 def full_rary_tree(r, n)

251 try: 

252 iter_wa = iter(which_args) 

253 except TypeError: 

254 iter_wa = (which_args,) 

255

356class argmap: 

357 """A decorator to apply a map to arguments before calling the function 

358 

359 This class provides a decorator that maps (transforms) arguments of the function 

360 before the function is called. Thus for example, we have similar code 

361 in many functions to determine whether an argument is the number of nodes 

362 to be created, or a list of nodes to be handled. The decorator provides 

363 the code to accept either -- transforming the indicated argument into a 

364 list of nodes before the actual function is called. 

365 

366 This decorator class allows us to process single or multiple arguments. 

367 The arguments to be processed can be specified by string, naming the argument, 

368 or by index, specifying the item in the args list. 

369 

370 Parameters 

371 ---------- 

372 func : callable 

373 The function to apply to arguments 

374 

375 *args : iterable of (int, str or tuple) 

376 A list of parameters, specified either as strings (their names), ints 

377 (numerical indices) or tuples, which may contain ints, strings, and 

378 (recursively) tuples. Each indicates which parameters the decorator 

379 should map. Tuples indicate that the map function takes (and returns) 

380 multiple parameters in the same order and nested structure as indicated 

381 here. 

382 

383 try_finally : bool (default: False) 

384 When True, wrap the function call in a try-finally block with code 

385 for the finally block created by `func`. This is used when the map 

386 function constructs an object (like a file handle) that requires 

387 post-processing (like closing). 

388 

389 Note: try_finally decorators cannot be used to decorate generator 

390 functions. 

391 

392 Examples 

393 -------- 

394 Most of these examples use `@argmap(...)` to apply the decorator to 

395 the function defined on the next line. 

396 In the NetworkX codebase however, `argmap` is used within a function to 

397 construct a decorator. That is, the decorator defines a mapping function 

398 and then uses `argmap` to build and return a decorated function. 

399 A simple example is a decorator that specifies which currency to report money. 

400 The decorator (named `convert_to`) would be used like:: 

401 

402 @convert_to("US_Dollars", "income") 

403 def show_me_the_money(name, income): 

404 print(f"{name} : {income}") 

405 

406 And the code to create the decorator might be:: 

407 

408 def convert_to(currency, which_arg): 

409 def _convert(amount): 

410 if amount.currency != currency: 

411 amount = amount.to_currency(currency) 

412 return amount 

413 return argmap(_convert, which_arg) 

414 

415 Despite this common idiom for argmap, most of the following examples 

416 use the `@argmap(...)` idiom to save space. 

417 

418 Here's an example use of argmap to sum the elements of two of the functions 

419 arguments. The decorated function:: 

420 

421 @argmap(sum, "xlist", "zlist") 

422 def foo(xlist, y, zlist): 

423 return xlist - y + zlist 

424 

425 is syntactic sugar for:: 

426 

427 def foo(xlist, y, zlist): 

428 x = sum(xlist) 

429 z = sum(zlist) 

430 return x - y + z 

431 

432 and is equivalent to (using argument indexes):: 

433 

434 @argmap(sum, "xlist", 2) 

435 def foo(xlist, y, zlist): 

436 return xlist - y + zlist 

437 

438 or:: 

439 

440 @argmap(sum, "zlist", 0) 

441 def foo(xlist, y, zlist): 

442 return xlist - y + zlist 

443 

444 Transforming functions can be applied to multiple arguments, such as:: 

445 

446 def swap(x, y): 

447 return y, x 

448 

449 # the 2-tuple tells argmap that the map `swap` has 2 inputs/outputs. 

450 @argmap(swap, ("a", "b")): 

451 def foo(a, b, c): 

452 return a / b * c 

453 

454 is equivalent to:: 

455 

456 def foo(a, b, c): 

457 a, b = swap(a, b) 

458 return a / b * c 

459 

460 More generally, the applied arguments can be nested tuples of strings or ints. 

461 The syntax `@argmap(some_func, ("a", ("b", "c")))` would expect `some_func` to 

462 accept 2 inputs with the second expected to be a 2-tuple. It should then return 

463 2 outputs with the second a 2-tuple. The returns values would replace input "a" 

464 "b" and "c" respectively. Similarly for `@argmap(some_func, (0, ("b", 2)))`. 

465 

466 Also, note that an index larger than the number of named parameters is allowed 

467 for variadic functions. For example:: 

468 

469 def double(a): 

470 return 2 * a 

471 

472 @argmap(double, 3) 

473 def overflow(a, *args): 

474 return a, args 

475 

476 print(overflow(1, 2, 3, 4, 5, 6)) # output is 1, (2, 3, 8, 5, 6) 

477 

478 **Try Finally** 

479 

480 Additionally, this `argmap` class can be used to create a decorator that 

481 initiates a try...finally block. The decorator must be written to return 

482 both the transformed argument and a closing function. 

483 This feature was included to enable the `open_file` decorator which might 

484 need to close the file or not depending on whether it had to open that file. 

485 This feature uses the keyword-only `try_finally` argument to `@argmap`. 

486 

487 For example this map opens a file and then makes sure it is closed:: 

488 

489 def open_file(fn): 

490 f = open(fn) 

491 return f, lambda: f.close() 

492 

493 The decorator applies that to the function `foo`:: 

494 

495 @argmap(open_file, "file", try_finally=True) 

496 def foo(file): 

497 print(file.read()) 

498 

499 is syntactic sugar for:: 

500 

501 def foo(file): 

502 file, close_file = open_file(file) 

503 try: 

504 print(file.read()) 

505 finally: 

506 close_file() 

507 

508 and is equivalent to (using indexes):: 

509 

510 @argmap(open_file, 0, try_finally=True) 

511 def foo(file): 

512 print(file.read()) 

513 

514 Here's an example of the try_finally feature used to create a decorator:: 

515 

516 def my_closing_decorator(which_arg): 

517 def _opener(path): 

518 if path is None: 

519 path = open(path) 

520 fclose = path.close 

521 else: 

522 # assume `path` handles the closing 

523 fclose = lambda: None 

524 return path, fclose 

525 return argmap(_opener, which_arg, try_finally=True) 

526 

527 which can then be used as:: 

528 

529 @my_closing_decorator("file") 

530 def fancy_reader(file=None): 

531 # this code doesn't need to worry about closing the file 

532 print(file.read()) 

533 

534 Decorators with try_finally = True cannot be used with generator functions, 

535 because the `finally` block is evaluated before the generator is exhausted:: 

536 

537 @argmap(open_file, "file", try_finally=True) 

538 def file_to_lines(file): 

539 for line in file.readlines(): 

540 yield line 

541 

542 is equivalent to:: 

543 

544 def file_to_lines_wrapped(file): 

545 for line in file.readlines(): 

546 yield line 

547 

548 def file_to_lines_wrapper(file): 

549 try: 

550 file = open_file(file) 

551 return file_to_lines_wrapped(file) 

552 finally: 

553 file.close() 

554 

555 which behaves similarly to:: 

556 

557 def file_to_lines_whoops(file): 

558 file = open_file(file) 

559 file.close() 

560 for line in file.readlines(): 

561 yield line 

562 

563 because the `finally` block of `file_to_lines_wrapper` is executed before 

564 the caller has a chance to exhaust the iterator. 

565 

566 Notes 

567 ----- 

568 An object of this class is callable and intended to be used when 

569 defining a decorator. Generally, a decorator takes a function as input 

570 and constructs a function as output. Specifically, an `argmap` object 

571 returns the input function decorated/wrapped so that specified arguments 

572 are mapped (transformed) to new values before the decorated function is called. 

573 

574 As an overview, the argmap object returns a new function with all the 

575 dunder values of the original function (like `__doc__`, `__name__`, etc). 

576 Code for this decorated function is built based on the original function's 

577 signature. It starts by mapping the input arguments to potentially new 

578 values. Then it calls the decorated function with these new values in place 

579 of the indicated arguments that have been mapped. The return value of the 

580 original function is then returned. This new function is the function that 

581 is actually called by the user. 

582 

583 Three additional features are provided. 

584 1) The code is lazily compiled. That is, the new function is returned 

585 as an object without the code compiled, but with all information 

586 needed so it can be compiled upon it's first invocation. This saves 

587 time on import at the cost of additional time on the first call of 

588 the function. Subsequent calls are then just as fast as normal. 

589 

590 2) If the "try_finally" keyword-only argument is True, a try block 

591 follows each mapped argument, matched on the other side of the wrapped 

592 call, by a finally block closing that mapping. We expect func to return 

593 a 2-tuple: the mapped value and a function to be called in the finally 

594 clause. This feature was included so the `open_file` decorator could 

595 provide a file handle to the decorated function and close the file handle 

596 after the function call. It even keeps track of whether to close the file 

597 handle or not based on whether it had to open the file or the input was 

598 already open. So, the decorated function does not need to include any 

599 code to open or close files. 

600 

601 3) The maps applied can process multiple arguments. For example, 

602 you could swap two arguments using a mapping, or transform 

603 them to their sum and their difference. This was included to allow 

604 a decorator in the `quality.py` module that checks that an input 

605 `partition` is a valid partition of the nodes of the input graph `G`. 

606 In this example, the map has inputs `(G, partition)`. After checking 

607 for a valid partition, the map either raises an exception or leaves 

608 the inputs unchanged. Thus many functions that make this check can 

609 use the decorator rather than copy the checking code into each function. 

610 More complicated nested argument structures are described below. 

611 

612 The remaining notes describe the code structure and methods for this 

613 class in broad terms to aid in understanding how to use it. 

614 

615 Instantiating an `argmap` object simply stores the mapping function and 

616 the input identifiers of which arguments to map. The resulting decorator 

617 is ready to use this map to decorate any function. Calling that object 

618 (`argmap.__call__`, but usually done via `@my_decorator`) a lazily 

619 compiled thin wrapper of the decorated function is constructed, 

620 wrapped with the necessary function dunder attributes like `__doc__` 

621 and `__name__`. That thinly wrapped function is returned as the 

622 decorated function. When that decorated function is called, the thin 

623 wrapper of code calls `argmap._lazy_compile` which compiles the decorated 

624 function (using `argmap.compile`) and replaces the code of the thin 

625 wrapper with the newly compiled code. This saves the compilation step 

626 every import of networkx, at the cost of compiling upon the first call 

627 to the decorated function. 

628 

629 When the decorated function is compiled, the code is recursively assembled 

630 using the `argmap.assemble` method. The recursive nature is needed in 

631 case of nested decorators. The result of the assembly is a number of 

632 useful objects. 

633 

634 sig : the function signature of the original decorated function as 

635 constructed by :func:`argmap.signature`. This is constructed 

636 using `inspect.signature` but enhanced with attribute 

637 strings `sig_def` and `sig_call`, and other information 

638 specific to mapping arguments of this function. 

639 This information is used to construct a string of code defining 

640 the new decorated function. 

641 

642 wrapped_name : a unique internally used name constructed by argmap 

643 for the decorated function. 

644 

645 functions : a dict of the functions used inside the code of this 

646 decorated function, to be used as `globals` in `exec`. 

647 This dict is recursively updated to allow for nested decorating. 

648 

649 mapblock : code (as a list of strings) to map the incoming argument 

650 values to their mapped values. 

651 

652 finallys : code (as a list of strings) to provide the possibly nested 

653 set of finally clauses if needed. 

654 

655 mutable_args : a bool indicating whether the `sig.args` tuple should be 

656 converted to a list so mutation can occur. 

657 

658 After this recursive assembly process, the `argmap.compile` method 

659 constructs code (as strings) to convert the tuple `sig.args` to a list 

660 if needed. It joins the defining code with appropriate indents and 

661 compiles the result. Finally, this code is evaluated and the original 

662 wrapper's implementation is replaced with the compiled version (see 

663 `argmap._lazy_compile` for more details). 

664 

665 Other `argmap` methods include `_name` and `_count` which allow internally 

666 generated names to be unique within a python session. 

667 The methods `_flatten` and `_indent` process the nested lists of strings 

668 into properly indented python code ready to be compiled. 

669 

670 More complicated nested tuples of arguments also allowed though 

671 usually not used. For the simple 2 argument case, the argmap 

672 input ("a", "b") implies the mapping function will take 2 arguments 

673 and return a 2-tuple of mapped values. A more complicated example 

674 with argmap input `("a", ("b", "c"))` requires the mapping function 

675 take 2 inputs, with the second being a 2-tuple. It then must output 

676 the 3 mapped values in the same nested structure `(newa, (newb, newc))`. 

677 This level of generality is not often needed, but was convenient 

678 to implement when handling the multiple arguments. 

679

697 @staticmethod 

698 def _lazy_compile(func): 

699 """Compile the source of a wrapped function 

700 

701 Assemble and compile the decorated function, and intrusively replace its 

702 code with the compiled version's. The thinly wrapped function becomes 

703 the decorated function. 

704 

705 Parameters 

706 ---------- 

707 func : callable 

708 A function returned by argmap.__call__ which is in the process 

709 of being called for the first time. 

710 

711 Returns 

712 ------- 

713 func : callable 

714 The same function, with a new __code__ object. 

715 

716 Notes 

717 ----- 

718 It was observed in NetworkX issue #4732 [1] that the import time of 

719 NetworkX was significantly bloated by the use of decorators: over half 

720 of the import time was being spent decorating functions. This was 

721 somewhat improved by a change made to the `decorator` library, at the 

722 cost of a relatively heavy-weight call to `inspect.Signature.bind` 

723 for each call to the decorated function. 

724 

725 The workaround we arrived at is to do minimal work at the time of 

726 decoration. When the decorated function is called for the first time, 

727 we compile a function with the same function signature as the wrapped 

728 function. The resulting decorated function is faster than one made by 

729 the `decorator` library, so that the overhead of the first call is 

730 'paid off' after a small number of calls. 

731 

732 References 

733 ---------- 

734 

735 [1] https://github.com/networkx/networkx/issues/4732 

736 

737 """ 

738 real_func = func.__argmap__.compile(func.__wrapped__) 

739 func.__code__ = real_func.__code__ 

740 func.__globals__.update(real_func.__globals__) 

741 func.__dict__.update(real_func.__dict__) 

742 return func

744 def __call__(self, f): 

745 """Construct a lazily decorated wrapper of f. 

746 

747 The decorated function will be compiled when it is called for the first time, 

748 and it will replace its own __code__ object so subsequent calls are fast. 

749 

750 Parameters 

751 ---------- 

752 f : callable 

753 A function to be decorated. 

754 

755 Returns 

756 ------- 

757 func : callable 

758 The decorated function. 

759 

760 See Also 

761 -------- 

762 argmap._lazy_compile 

763 """ 

764 

765 def func(*args, __wrapper=None, **kwargs): 

766 return argmap._lazy_compile(__wrapper)(*args, **kwargs) 

767 

768 # standard function-wrapping stuff 

769 func.__name__ = f.__name__ 

770 func.__doc__ = f.__doc__ 

771 func.__defaults__ = f.__defaults__ 

772 func.__kwdefaults__.update(f.__kwdefaults__ or {}) 

773 func.__module__ = f.__module__ 

774 func.__qualname__ = f.__qualname__ 

775 func.__dict__.update(f.__dict__) 

776 func.__wrapped__ = f 

777 

778 # now that we've wrapped f, we may have picked up some __dict__ or 

779 # __kwdefaults__ items that were set by a previous argmap. Thus, we set 

780 # these values after those update() calls. 

781 

782 # If we attempt to access func from within itself, that happens through 

783 # a closure -- which trips an error when we replace func.__code__. The 

784 # standard workaround for functions which can't see themselves is to use 

785 # a Y-combinator, as we do here. 

786 func.__kwdefaults__["_argmap__wrapper"] = func 

787 

788 # this self-reference is here because functools.wraps preserves 

789 # everything in __dict__, and we don't want to mistake a non-argmap 

790 # wrapper for an argmap wrapper 

791 func.__self__ = func 

792 

793 # this is used to variously call self.assemble and self.compile 

794 func.__argmap__ = self 

795 

796 if hasattr(f, "__argmap__"): 

797 func.__is_generator = f.__is_generator 

798 else: 

799 func.__is_generator = inspect.isgeneratorfunction(f) 

800 

801 if self._finally and func.__is_generator: 

802 raise nx.NetworkXError("argmap cannot decorate generators with try_finally") 

803 

804 return func

851 def compile(self, f): 

852 """Compile the decorated function. 

853 

854 Called once for a given decorated function -- collects the code from all 

855 argmap decorators in the stack, and compiles the decorated function. 

856 

857 Much of the work done here uses the `assemble` method to allow recursive 

858 treatment of multiple argmap decorators on a single decorated function. 

859 That flattens the argmap decorators, collects the source code to construct 

860 a single decorated function, then compiles/executes/returns that function. 

861 

862 The source code for the decorated function is stored as an attribute 

863 `_code` on the function object itself. 

864 

865 Note that Python's `compile` function requires a filename, but this 

866 code is constructed without a file, so a fictitious filename is used 

867 to describe where the function comes from. The name is something like: 

868 "argmap compilation 4". 

869 

870 Parameters 

871 ---------- 

872 f : callable 

873 The function to be decorated 

874 

875 Returns 

876 ------- 

877 func : callable 

878 The decorated file 

879 

880 """ 

881 sig, wrapped_name, functions, mapblock, finallys, mutable_args = self.assemble( 

882 f 

883 ) 

884 

885 call = f"{sig.call_sig.format(wrapped_name)}#" 

886 mut_args = f"{sig.args} = list({sig.args})" if mutable_args else "" 

887 body = argmap._indent(sig.def_sig, mut_args, mapblock, call, finallys) 

888 code = "\n".join(body) 

889 

890 locl = {} 

891 globl = dict(functions.values()) 

892 filename = f"{self.__class__} compilation {self._count()}" 

893 compiled = compile(code, filename, "exec") 

894 exec(compiled, globl, locl) 

895 func = locl[sig.name] 

896 func._code = code 

897 return func

899 def assemble(self, f): 

900 """Collects components of the source for the decorated function wrapping f. 

901 

902 If `f` has multiple argmap decorators, we recursively assemble the stack of 

903 decorators into a single flattened function. 

904 

905 This method is part of the `compile` method's process yet separated 

906 from that method to allow recursive processing. The outputs are 

907 strings, dictionaries and lists that collect needed info to 

908 flatten any nested argmap-decoration. 

909 

910 Parameters 

911 ---------- 

912 f : callable 

913 The function to be decorated. If f is argmapped, we assemble it. 

914 

915 Returns 

916 ------- 

917 sig : argmap.Signature 

918 The function signature as an `argmap.Signature` object. 

919 wrapped_name : str 

920 The mangled name used to represent the wrapped function in the code 

921 being assembled. 

922 functions : dict 

923 A dictionary mapping id(g) -> (mangled_name(g), g) for functions g 

924 referred to in the code being assembled. These need to be present 

925 in the ``globals`` scope of ``exec`` when defining the decorated 

926 function. 

927 mapblock : list of lists and/or strings 

928 Code that implements mapping of parameters including any try blocks 

929 if needed. This code will precede the decorated function call. 

930 finallys : list of lists and/or strings 

931 Code that implements the finally blocks to post-process the 

932 arguments (usually close any files if needed) after the 

933 decorated function is called. 

934 mutable_args : bool 

935 True if the decorator needs to modify positional arguments 

936 via their indices. The compile method then turns the argument 

937 tuple into a list so that the arguments can be modified. 

938 """ 

939 

940 # first, we check if f is already argmapped -- if that's the case, 

941 # build up the function recursively. 

942 # > mapblock is generally a list of function calls of the sort 

943 # arg = func(arg) 

944 # in addition to some try-blocks if needed. 

945 # > finallys is a recursive list of finally blocks of the sort 

946 # finally: 

947 # close_func_1() 

948 # finally: 

949 # close_func_2() 

950 # > functions is a dict of functions used in the scope of our decorated 

951 # function. It will be used to construct globals used in compilation. 

952 # We make functions[id(f)] = name_of_f, f to ensure that a given 

953 # function is stored and named exactly once even if called by 

954 # nested decorators. 

955 if hasattr(f, "__argmap__") and f.__self__ is f: 

956 ( 

957 sig, 

958 wrapped_name, 

959 functions, 

960 mapblock, 

961 finallys, 

962 mutable_args, 

963 ) = f.__argmap__.assemble(f.__wrapped__) 

964 functions = dict(functions) # shallow-copy just in case 

965 else: 

966 sig = self.signature(f) 

967 wrapped_name = self._name(f) 

968 mapblock, finallys = [], [] 

969 functions = {id(f): (wrapped_name, f)} 

970 mutable_args = False 

971 

972 if id(self._func) in functions: 

973 fname, _ = functions[id(self._func)] 

974 else: 

975 fname, _ = functions[id(self._func)] = self._name(self._func), self._func 

976 

977 # this is a bit complicated -- we can call functions with a variety of 

978 # nested arguments, so long as their input and output are tuples with 

979 # the same nested structure. e.g. ("a", "b") maps arguments a and b. 

980 # A more complicated nesting like (0, (3, 4)) maps arguments 0, 3, 4 

981 # expecting the mapping to output new values in the same nested shape. 

982 # The ability to argmap multiple arguments was necessary for 

983 # the decorator `nx.algorithms.community.quality.require_partition`, and 

984 # while we're not taking full advantage of the ability to handle 

985 # multiply-nested tuples, it was convenient to implement this in 

986 # generality because the recursive call to `get_name` is necessary in 

987 # any case. 

988 applied = set() 

989 

990 def get_name(arg, first=True): 

991 nonlocal mutable_args 

992 if isinstance(arg, tuple): 

993 name = ", ".join(get_name(x, False) for x in arg) 

994 return name if first else f"({name})" 

995 if arg in applied: 

996 raise nx.NetworkXError(f"argument {arg} is specified multiple times") 

997 applied.add(arg) 

998 if arg in sig.names: 

999 return sig.names[arg] 

1000 elif isinstance(arg, str): 

1001 if sig.kwargs is None: 

1002 raise nx.NetworkXError( 

1003 f"name {arg} is not a named parameter and this function doesn't have kwargs" 

1004 ) 

1005 return f"{sig.kwargs}[{arg!r}]" 

1006 else: 

1007 if sig.args is None: 

1008 raise nx.NetworkXError( 

1009 f"index {arg} not a parameter index and this function doesn't have args" 

1010 ) 

1011 mutable_args = True 

1012 return f"{sig.args}[{arg - sig.n_positional}]" 

1013 

1014 if self._finally: 

1015 # here's where we handle try_finally decorators. Such a decorator 

1016 # returns a mapped argument and a function to be called in a 

1017 # finally block. This feature was required by the open_file 

1018 # decorator. The below generates the code 

1019 # 

1020 # name, final = func(name) #<--append to mapblock 

1021 # try: #<--append to mapblock 

1022 # ... more argmapping and try blocks 

1023 # return WRAPPED_FUNCTION(...) 

1024 # ... more finally blocks 

1025 # finally: #<--prepend to finallys 

1026 # final() #<--prepend to finallys 

1027 # 

1028 for a in self._args: 

1029 name = get_name(a) 

1030 final = self._name(name) 

1031 mapblock.append(f"{name}, {final} = {fname}({name})") 

1032 mapblock.append("try:") 

1033 finallys = ["finally:", f"{final}()#", "#", finallys] 

1034 else: 

1035 mapblock.extend( 

1036 f"{name} = {fname}({name})" for name in map(get_name, self._args) 

1037 ) 

1038 

1039 return sig, wrapped_name, functions, mapblock, finallys, mutable_args

1041 @classmethod 

1042 def signature(cls, f): 

1043 r"""Construct a Signature object describing `f` 

1044 

1045 Compute a Signature so that we can write a function wrapping f with 

1046 the same signature and call-type. 

1047 

1048 Parameters 

1049 ---------- 

1050 f : callable 

1051 A function to be decorated 

1052 

1053 Returns 

1054 ------- 

1055 sig : argmap.Signature 

1056 The Signature of f 

1057 

1058 Notes 

1059 ----- 

1060 The Signature is a namedtuple with names: 

1061 

1062 name : a unique version of the name of the decorated function 

1063 signature : the inspect.signature of the decorated function 

1064 def_sig : a string used as code to define the new function 

1065 call_sig : a string used as code to call the decorated function 

1066 names : a dict keyed by argument name and index to the argument's name 

1067 n_positional : the number of positional arguments in the signature 

1068 args : the name of the VAR_POSITIONAL argument if any, i.e. \*theseargs 

1069 kwargs : the name of the VAR_KEYWORDS argument if any, i.e. \*\*kwargs 

1070 

1071 These named attributes of the signature are used in `assemble` and `compile` 

1072 to construct a string of source code for the decorated function. 

1073 

1074 """ 

1075 sig = inspect.signature(f, follow_wrapped=False) 

1076 def_sig = [] 

1077 call_sig = [] 

1078 names = {} 

1079 

1080 kind = None 

1081 args = None 

1082 kwargs = None 

1083 npos = 0 

1084 for i, param in enumerate(sig.parameters.values()): 

1085 # parameters can be position-only, keyword-or-position, keyword-only 

1086 # in any combination, but only in the order as above. we do edge 

1087 # detection to add the appropriate punctuation 

1088 prev = kind 

1089 kind = param.kind 

1090 if prev == param.POSITIONAL_ONLY != kind: 

1091 # the last token was position-only, but this one isn't 

1092 def_sig.append("/") 

1093 if prev != param.KEYWORD_ONLY == kind != param.VAR_POSITIONAL: 

1094 # param is the first keyword-only arg and isn't starred 

1095 def_sig.append("*") 

1096 

1097 # star arguments as appropriate 

1098 if kind == param.VAR_POSITIONAL: 

1099 name = "*" + param.name 

1100 args = param.name 

1101 count = 0 

1102 elif kind == param.VAR_KEYWORD: 

1103 name = "**" + param.name 

1104 kwargs = param.name 

1105 count = 0 

1106 else: 

1107 names[i] = names[param.name] = param.name 

1108 name = param.name 

1109 count = 1 

1110 

1111 # assign to keyword-only args in the function call 

1112 if kind == param.KEYWORD_ONLY: 

1113 call_sig.append(f"{name} = {name}") 

1114 else: 

1115 npos += count 

1116 call_sig.append(name) 

1117 

1118 def_sig.append(name) 

1119 

1120 fname = cls._name(f) 

1121 def_sig = f'def {fname}({", ".join(def_sig)}):' 

1122 

1123 call_sig = f"return {{}}({', '.join(call_sig)})" 

1124 

1125 return cls.Signature(fname, sig, def_sig, call_sig, names, npos, args, kwargs)

1142 def _flatten(nestlist, visited): 

1143 """flattens a recursive list of lists that doesn't have cyclic references 

1144 

1145 Parameters 

1146 ---------- 

1147 nestlist : iterable 

1148 A recursive list of objects to be flattened into a single iterable 

1149 

1150 visited : set 

1151 A set of object ids which have been walked -- initialize with an 

1152 empty set 

1153 

1154 Yields 

1155 ------

1157 

1158 """

1162 raise ValueError("A cycle was found in nestlist. Be a tree.")
